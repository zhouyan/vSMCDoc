366  void discard(result_type nskip)
367  {
368  std::size_t n = static_cast<std::size_t>(nskip);
369  if (index_ + n <= K_) {
370  index_ += n;
371  return;
372  }
373 
374  n -= K_ - index_;
375  if (n <= K_) {
376  index_ = K_;
377  operator()();
378  index_ = n;
379  return;
380  }
381

383  index_ = K_;
384  operator()();
385  index_ = n % K_;
386  }
