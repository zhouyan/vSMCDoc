241  void discard(result_type nskip)
242  {
243  std::size_t n = static_cast<std::size_t>(nskip);
244  if (index_ + n <= M_) {
245  index_ += n;
246  return;
247  }
248 
249  n -= M_ - index_;
250  if (n <= M_) {
251  index_ = M_;
252  operator()();

258  index_ = M_;

261  }

336  {
