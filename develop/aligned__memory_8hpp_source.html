107 class AlignedMemorySTD
108 {
109  public:
110  static void *aligned_malloc(std::size_t n, std::size_t alignment)
111  {
112  VSMC_RUNTIME_ASSERT_UTILITY_ALIGNED_MEMORY;
113 
114  if (n == 0)
115  return nullptr;
116 
117  void *orig_ptr = std::malloc(n + alignment + sizeof(void *));
118  if (orig_ptr == nullptr)
119  throw std::bad_alloc();
120 
121  uintptr_t address = reinterpret_cast<uintptr_t>(orig_ptr);
122  uintptr_t offset = alignment - (address + sizeof(void *)) % alignment;
123  void *ptr =
124  reinterpret_cast<void *>(address + offset + sizeof(void *));
125  void **orig = reinterpret_cast<void **>(address + offset);
126  *orig = orig_ptr;
127 
128  return ptr;
129  }
130 
131  static void aligned_free(void *ptr)
132  {
133  std::free(*reinterpret_cast<void **>(
