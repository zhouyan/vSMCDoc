173 class AlignedMemorySTD
174 {
175  public:
176  static void *aligned_malloc(std::size_t n, std::size_t alignment) noexcept
177  {
178  void *orig_ptr =
179  std::malloc((n > 0 ? n : 1) + alignment + sizeof(void *));
180  if (orig_ptr == nullptr)
181  return nullptr;
182 
183  uintptr_t address = reinterpret_cast<uintptr_t>(orig_ptr);
184  uintptr_t offset = alignment - (address + sizeof(void *)) % alignment;
185  void *ptr =
186  reinterpret_cast<void *>(address + offset + sizeof(void *));
187  void **orig = reinterpret_cast<void **>(address + offset);
188  *orig = orig_ptr;
189 
190  return ptr;
191  }
192 
193  static void aligned_free(void *ptr) noexcept
194  {
195  if (ptr != nullptr) {
196  std::free(*reinterpret_cast<void **>(
197  reinterpret_cast<uintptr_t>(ptr) - sizeof(void *)));
198  }
199  }
200 }; // class AlignedMemmorySTD
