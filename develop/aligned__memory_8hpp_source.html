166 class AlignedMemorySTD
167 {
168  public:
169  static void *aligned_malloc(std::size_t n, std::size_t alignment) noexcept
170  {
171  void *orig_ptr =
172  std::malloc((n > 0 ? n : 1) + alignment + sizeof(void *));
173  if (orig_ptr == nullptr)
174  return nullptr;
175 
176  uintptr_t address = reinterpret_cast<uintptr_t>(orig_ptr);
177  uintptr_t offset = alignment - (address + sizeof(void *)) % alignment;
178  void *ptr =
179  reinterpret_cast<void *>(address + offset + sizeof(void *));
180  void **orig = reinterpret_cast<void **>(address + offset);
181  *orig = orig_ptr;
182 
183  return ptr;
184  }
185 
186  static void aligned_free(void *ptr) noexcept
187  {
188  if (ptr != nullptr) {
189  std::free(*reinterpret_cast<void **>(
190  reinterpret_cast<uintptr_t>(ptr) - sizeof(void *)));
191  }
192  }
193 }; // class AlignedMemmorySTD
