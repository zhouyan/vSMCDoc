413  template <typename CharT, typename Traits>
414  void build_program(
415  const std::string flags, std::basic_ostream<CharT, Traits> &os)
416  {
417  ++build_id_;
418 
419  build_ = false;
420  ::cl_int status = program_.build(manager().device_vec(), flags);
421  if (status != CL_SUCCESS) {
422  std::vector<CLDevice> dev_vec(program_.get_device());
423  std::string equal(75, '=');
424  std::string dash(75, '-');
425  std::string name;
426  for (const auto &dev : dev_vec) {
427  dev.get_info(CL_DEVICE_NAME, name);
428  os << equal << std::endl;
429  if (program_.build_status(dev) == CL_BUILD_SUCCESS)
430  os << "Build success for " << name << std::endl;
431  else
432  os << "Build failure for " << name << std::endl;
433  os << dash << std::endl;
434  os << program_.build_log(dev) << std::endl;
435  }
436  os << equal << std::endl;
437  return;
438  }
439  copy_.build(size_, state_size_);
440  build_ = true;
441  }
