169  offset_ += size_all_[i];
170  global_size_ += size_all_[i];
171  size_equal_ = size_equal_ && N == size_all_[i];
172  }
173  for (std::size_t i = R; i != S; ++i) {
174  global_size_ += size_all_[i];
175  size_equal_ = size_equal_ && N == size_all_[i];
176  }

270  void copy_this_node(size_type N, const size_type *src_idx,
271  std::vector<std::pair<int, size_type>> &copy_recv,
272  std::vector<std::pair<int, size_type>> &copy_send)
273  {
274  using std::advance;
275 
276  const int rank_this = world_.rank();
277 
278  src_idx_this_.resize(this->size());
279  const size_type *first = src_idx + offset_;
280  for (size_type dst = 0; dst != this->size(); ++dst, ++first) {
281  size_type src = *first;
282  src_idx_this_[dst] = rank_this == rank(src) ? local_id(src) : dst;
283  }
284  StateBase::copy(this->size(), src_idx_this_.data());

286  copy_recv.clear();
287  copy_send.clear();
288  for (size_type dst = 0; dst != N; ++dst) {
289  size_type src = src_idx[dst];
290  int rank_recv = rank(dst);
291  int rank_send = rank(src);
292  size_type id_recv = local_id(dst);
293  size_type id_send = local_id(src);
294  if (rank_this == rank_recv && rank_this == rank_send) {
295  continue;
296  } else if (rank_this == rank_recv) {
297  copy_recv.push_back(std::make_pair(rank_send, id_recv));
298  } else if (rank_this == rank_send) {
299  copy_send.push_back(std::make_pair(rank_recv, id_send));

309  const std::vector<std::pair<int, size_type>> &copy_recv,
310  const std::vector<std::pair<int, size_type>> &copy_send)
311  {
312  const int rank_this = world_.rank();
313  for (int r = 0; r != world_.size(); ++r) {
314  if (rank_this == r) {

317  this->state_unpack(
