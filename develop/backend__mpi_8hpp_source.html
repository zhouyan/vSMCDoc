123  double compute_ess(const double *first, bool use_log) const
124  {
125  const std::size_t N = static_cast<std::size_t>(this->size());
126  const double *const wptr = this->data();
127 
128  Vector<double> buffer(N);
129  if (use_log) {
130  math::vLn(N, wptr, buffer.data());
131  math::vAdd(N, buffer.data(), first, buffer.data());
132  double lwmax = *(std::max_element(buffer.begin(), buffer.end()));
133  double gwmax = 0;
134  ::boost::mpi::all_reduce(
135  world_, lwmax, gwmax, ::boost::mpi::maximum<double>());
136  for (std::size_t i = 0; i != N; ++i)
137  buffer[i] -= gwmax;
138  math::vExp(N, buffer.data(), buffer.data());
139  } else {

141  }

144  double gcoeff = 0;

146  math::scal(N, 1 / gcoeff, buffer.data(), 1);
147

155  double compute_cess(const double *first, bool use_log) const
156  {
157  const std::size_t N = static_cast<std::size_t>(this->size());
158  const double *const wptr = this->data();
159 
160  Vector<double> buffer;
161  if (use_log) {
162  buffer.resize(N);
163  math::vExp(N, first, buffer.data());
164  }
165  const double *const bptr = use_log ? buffer.data() : first;
166 
167  double labove = 0;
168  double lbelow = 0;
169  for (std::size_t i = 0; i != N; ++i) {
170  double wb = wptr[i] * bptr[i];
171  labove += wb;
172  lbelow += wb * bptr[i];
173  }
174  double gabove = 0;
175  double gbelow = 0;
176  ::boost::mpi::all_reduce(world_, labove, gabove, std::plus<double>());
177  ::boost::mpi::all_reduce(world_, lbelow, gbelow, std::plus<double>());
178 
179  return gabove * gabove / gbelow;
180  }

396  void copy_this_node(size_type N, const size_type *src_idx,
397  std::vector<std::pair<int, size_type>> &copy_recv,
398  std::vector<std::pair<int, size_type>> &copy_send)
