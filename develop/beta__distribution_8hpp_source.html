69 template <typename RealType>
70 inline BetaDistributionAlgorithm beta_distribution_algorithm(
71  RealType alpha, RealType beta)
72 {
73  const RealType K = static_cast<RealType>(0.852);
74  const RealType C = static_cast<RealType>(-0.956);
75  const RealType D = beta + K * alpha * alpha + C;
76  if (is_equal<RealType>(alpha, 0.5) && is_equal<RealType>(beta, 0.5))
77  return BetaDistributionAlgorithmAS;
78  else if (is_equal<RealType>(alpha, 1) && is_equal<RealType>(beta, 1))
79  return BetaDistributionAlgorithm11;
80  else if (is_equal<RealType>(alpha, 1))
81  return BetaDistributionAlgorithm1X;
82  else if (is_equal<RealType>(beta, 1))
83  return BetaDistributionAlgorithmX1;
84  else if (alpha > 1 && beta > 1)
85  return BetaDistributionAlgorithmC;
86  else if (alpha < 1 && beta < 1 && D <= 0)
87  return BetaDistributionAlgorithmJ;
88  else if (alpha < 1 && beta < 1)
89  return BetaDistributionAlgorithmA1;
90  else if (alpha < 1 && beta > 1)
91  return BetaDistributionAlgorithmA2;
92  else if (alpha > 1 && beta < 1)
93  return BetaDistributionAlgorithmA3;
94  return BetaDistributionAlgorithmC;
95 }

97 template <typename RealType>
98 class BetaDistributionConstant
99 {
100  public:
101  BetaDistributionConstant(RealType alpha, RealType beta)
102  : algorithm(beta_distribution_algorithm(alpha, beta))
103  , a(0)
104  , b(0)
105  , t(0)
106  , p(0)
107  {
108  switch (algorithm) {
109  case BetaDistributionAlgorithmAS: break;
110  case BetaDistributionAlgorithm11: break;
111  case BetaDistributionAlgorithm1X: b = 1 / beta; break;
112  case BetaDistributionAlgorithmX1: a = 1 / alpha; break;
113  case BetaDistributionAlgorithmC:
114  a = alpha + beta;
115  b = std::min(alpha, beta);
116  if (b > 1)
117  b = std::sqrt((2 * alpha * beta - a) / (a - 2));
118  b = 1 / b;
119  t = alpha + 1 / b;
120  p = a * std::log(a);
121  break;
122  case BetaDistributionAlgorithmJ:
123  a = 1 / alpha;
124  b = 1 / beta;
125  break;
126  case BetaDistributionAlgorithmA1:
127  a = 1 / alpha;
128  b = 1 / beta;
129  t = std::sqrt(alpha * (1 - alpha));
130  t /= t + std::sqrt(beta * (1 - beta));
131  p = beta * t;
132  p /= p + alpha * (1 - t);
133  break;
134  case BetaDistributionAlgorithmA2:
135  a = 1 / alpha;
136  b = 1 / beta;
137  t = 1 - alpha;
138  t /= t + beta;
139  p = beta * t;
140  p /= p + alpha * std::pow(1 - t, beta);
141  break;
142  case BetaDistributionAlgorithmA3:
143  a = 1 / beta;
144  b = 1 / alpha;
145  t = 1 - beta;
146  t /= t + alpha;
147  p = alpha * t;
148  p /= p + beta * std::pow(1 - t, alpha);
149  break;
150  }
151  }
152 
153  BetaDistributionAlgorithm algorithm;
154  RealType a;
155  RealType b;
156  RealType t;
157  RealType p;
158 }; // class BetaDistributionConstant

350  template <typename RNGType>
351  result_type generate_c(RNGType &rng)
352  {
353  U01CODistribution<RealType> runif;
354  const result_type ln_4 = 2 * const_ln_2<result_type>();
355  result_type x = 0;
356  result_type y = 0;
357  result_type left = 0;
358  result_type right = 0;
359  do {
360  result_type u1 = runif(rng);
361  result_type u2 = runif(rng);
362  result_type v = param_.constant_.b * std::log(u1 / (1 - u1));
363  x = param_.alpha_ * std::exp(v);
364  y = param_.beta_ + x;
365  left = (param_.constant_.p - param_.constant_.a * std::log(y)) +
366  (param_.constant_.t * v - ln_4);
367  right = std::log(u1 * u1 * u2);
368  } while (left < right);
369 
370  return x / y;
371  }

387  template <typename RNGType>
388  result_type generate_a1(RNGType &rng)
389  {
390  U01CODistribution<RealType> runif;
391  while (true) {
392  result_type u = runif(rng);
393  result_type e = -std::log(runif(rng));
394  result_type x = 0;
395  result_type v = 0;
396  if (u < param_.constant_.p) {
397  x = param_.constant_.t *
398  std::pow(u / param_.constant_.p, param_.constant_.a);
399  v = (1 - param_.beta_) *
400  std::log((1 - x) / (1 - param_.constant_.t));
401  } else {
402  x = 1 -
403  (1 - param_.constant_.t) *
404  std::pow((1 - u) / (1 - param_.constant_.p),
405  param_.constant_.b);
406  v = (1 - param_.alpha_) * std::log(x / param_.constant_.t);
407  }
408  if (v < e)
409  return x;
410  }
411  }

517 template <std::size_t K, typename RealType, typename RNGType>
518 inline std::size_t beta_distribution_impl_c(RNGType &rng, std::size_t n,
519  RealType *r, RealType alpha, RealType beta,
520  const BetaDistributionConstant<RealType> &constant)
521 {
522  const RealType a = constant.a;
523  const RealType b = constant.b;
524  const RealType t = constant.t;
525  const RealType p = constant.p;
526  const RealType ln_4 = 2 * const_ln_2<RealType>();
527  RealType s[K * 5];
528  RealType *const u1 = s;
529  RealType *const u2 = s + n;
530  RealType *const v = s + n * 2;
531  RealType *const x = s + n * 3;
532  RealType *const y = s + n * 4;
533  u01_co_distribution(rng, n * 2, s);
534  sub(n, static_cast<RealType>(1), u1, v);
535  div(n, u1, v, v);
536  log(n, v, v);
537  mul(n, b, v, v);
538  exp(n, v, x);
539  mul(n, alpha, x, x);
540  add(n, beta, x, y);
541  div(n, x, y, x);
542  sqr(n, u1, u1);
543  mul(n, u1, u2, u2);
544  log(n, u2, u2);
545  log(n, y, u1);
546  fma(n, p, -a, u1, u1);
547  fma(n, -ln_4, t, v, v);
548  add(n, v, u1, u1);
549 
550  std::size_t m = 0;
551  for (std::size_t i = 0; i != n; ++i)
552  if (u1[i] > u2[i])
553  r[m++] = x[i];
554 
555  return m;
556 }

645 template <typename RealType, typename RNGType>
646 inline void beta_distribution(
647  RNGType &rng, std::size_t n, RealType *r, RealType alpha, RealType beta)
648 {
649  const std::size_t k = 1000;
650  const internal::BetaDistributionConstant<RealType> constant(alpha, beta);
651  while (n > k) {
652  std::size_t m = internal::beta_distribution_impl<k>(
653  rng, k, r, alpha, beta, constant);
654  if (m == 0)
655  break;
656  n -= m;
657  r += m;
658  }
659  std::size_t m =
660  internal::beta_distribution_impl<k>(rng, n, r, alpha, beta, constant);
661  n -= m;
662  r += m;
663  if (n > 0) {
664  BetaDistribution<RealType> dist(alpha, beta);
665  for (std::size_t i = 0; i != n; ++i)
666  r[i] = dist(rng);
667  }
