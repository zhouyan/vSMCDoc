134 template <typename T>
135 inline void gemv(MatrixOrder order, MatrixTrans trans, std::size_t m,
136  std::size_t n, T alpha, const T *A, std::size_t lda, const T *x,
137  std::size_t incx, T beta, T *y, std::size_t incy)
138 {
139  std::size_t nrow = trans == NoTrans ? m : n;
140  std::size_t ncol = trans == NoTrans ? n : m;
141 
142  scal(nrow, beta, y, incy);
143 
144  if ((order == RowMajor && trans == NoTrans) ||
145  (order == ColMajor && trans == Trans)) {
146  std::size_t k = 0;
147  for (std::size_t r = 0; r != nrow; ++r, k += incy)
148  y[k] += alpha * dot<T>(ncol, x, incx, A + r * lda, 1);
149  } else {
150  std::size_t j = 0;
151  for (std::size_t c = 0; c != ncol; ++c, j += incx) {
152  std::size_t k = 0;
153  std::size_t l = c * lda;
154  const double ax = alpha * x[j];
155  for (std::size_t r = 0; r != nrow; ++r, ++l, k += incy)
156  y[k] += ax * A[l];
157  }
158  }
159 }
