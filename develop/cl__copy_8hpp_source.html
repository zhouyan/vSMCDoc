70  void build(std::size_t size, std::size_t state_size)
71  {
72  size_ = size;
73 
74  std::stringstream ss;
75 
76  ss << "#define Size " << size << "UL\n";
77  ss << "typedef struct {char c[" << state_size << "];} state_type;";
78 
79  ss << "__kernel void copy (__global const ulong *copy_from,\n";
80  ss << " __global state_type *state)\n";
81  ss << "{\n";
82  ss << " ulong to = get_global_id(0);\n";
83  ss << " if (to >= Size) return;\n";
84  ss << " ulong from = copy_from[to];\n";
85  ss << " if (to == from) return;\n";
86  ss << " state[to] = state[from];\n";
87  ss << "}\n";
88 
89  ss << "__kernel void copy_post (__global const char *idx,\n";
90  ss << " __global const state_type *tmp,\n";
91  ss << " __global state_type *state)\n";
92  ss << "{\n";
93  ss << " ulong id = get_global_id(0);\n";
94  ss << " if (id >= Size) return;\n";
95  ss << " if (idx[id] != 0) state[id] = tmp[id];\n";
96  ss << "}\n";
97 
98  ::cl_int status = CL_SUCCESS;
99 
100  program_ = manager().create_program(ss.str());
101 
102  std::vector<::cl_device_id> dev_vec_ptr;
103  for (const auto &dev : manager().device_vec())
104  dev_vec_ptr.push_back(dev.get());
105  status = ::clBuildProgram(program_.get(),
106  static_cast<::cl_uint>(dev_vec_ptr.size()), dev_vec_ptr.data(),
107  nullptr, nullptr, nullptr);
108  internal::cl_error_check(status, "CLCopy::build", "::clBuildProgram");
109 
110  ::cl_kernel kern = nullptr;
111 
112  kern = ::clCreateKernel(program_.get(), "copy", &status);
113  internal::cl_error_check(status, "CLCopy::build", "::clCreateKernel");
114  kernel_ = cl_kernel_make_shared(kern);
115 
116  kern = ::clCreateKernel(program_.get(), "copy_post", &status);
117  internal::cl_error_check(status, "CLCopy::build", "::clCreateKernel");
118  kernel_post_ = cl_kernel_make_shared(kern);
119 
120  configure_.local_size(size, kernel_.get(), manager().device().get());
121  configure_post_.local_size(
122  size, kernel_post_.get(), manager().device().get());
123  }
