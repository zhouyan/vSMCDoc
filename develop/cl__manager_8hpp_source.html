419  void setup_cl_manager(::cl_device_type dev_type)
420  {
421  setup_ = false;
422 
423  if (!platform_filter(dev_type)) {
424  VSMC_RUNTIME_WARNING_OPENCL_CL_MANAGER_SETUP_PLATFORM;
425  return;
426  }
427 
428  std::vector<CLDevice> dev_pool(platform_.get_device(dev_type));
429  std::vector<CLDevice> dev_select(device_filter(dev_pool));
430  if (dev_select.size() == 0) {
431  VSMC_RUNTIME_WARNING_OPENCL_CL_MANAGER_SETUP_CONTEXT;
432  VSMC_RUNTIME_WARNING_OPENCL_CL_MANAGER_SETUP_DEVICE;
433  return;
434  }
435 
436  ::cl_context_properties properties[] = {CL_CONTEXT_PLATFORM,
437  reinterpret_cast<::cl_context_properties>(platform_.get()), 0};
438  context_ = CLContext(properties, dev_select);
439  if (!bool(context_)) {
440  VSMC_RUNTIME_WARNING_OPENCL_CL_MANAGER_SETUP_CONTEXT;
441  return;
442  }
443 
444  device_vec_ = context_.get_device();
445  if (device_vec_.empty()) {
446  VSMC_RUNTIME_WARNING_OPENCL_CL_MANAGER_SETUP_DEVICE;
447  return;
448  }
449  device_ = device_vec_.front();
450 
451  command_queue_ = CLCommandQueue(context_, device_, 0);
452  if (!bool(command_queue_)) {
453  VSMC_RUNTIME_WARNING_OPENCL_CL_MANAGER_SETUP_COMMAND_QUEUE;
454  return;
455  }
456 
457  check_opencl_version();
458 
459  setup_ = true;
460  }

462  bool platform_filter(::cl_device_type dev_type)
463  {
464  std::vector<CLPlatform> plat_vec(CLPlatform::platforms());
465  if (plat_vec.empty())
466  return false;
467 
468  // If not using default platform
469  if (!setup_default_.default_platform()) {
470  for (const auto &plat : plat_vec) {
471  std::string name;
472  plat.get_info(CL_PLATFORM_NAME, name);
473  if (setup_default_.check_platform(name)) {
474  platform_ = plat;
475  return true;
476  }
477  }
478 
479  return false;
480  }
481 
482  // Using default platform: finding the first that has the device type
483  for (const auto &plat : plat_vec) {
484  std::vector<CLDevice> dev_pool(plat.get_device(dev_type));
485  std::vector<CLDevice> dev_select(device_filter(dev_pool));
486  if (!dev_select.empty()) {
487  platform_ = plat;
488  return true;
489  }
490  }
491 
492  return false;
493  }

495  std::vector<CLDevice> device_filter(const std::vector<CLDevice> &dev_pool)
496  {
497  std::vector<bool> dev_select_idx(dev_pool.size(), true);
498  std::vector<bool> dev_select_idx_igpu(dev_pool.size(), true);
499  std::string name;
500  ::cl_device_type type;
501 
502  // Filter out Intel GPU
503  for (std::size_t d = 0; d != dev_pool.size(); ++d) {
504  dev_pool[d].get_info(CL_DEVICE_TYPE, type);
505  dev_pool[d].get_info(CL_DEVICE_VENDOR, name);
506  if ((type & CL_DEVICE_TYPE_GPU) != 0 &&
507  name.find("Intel") != std::string::npos) {
508  dev_select_idx_igpu[d] = false;
509  }
510  }
511 
512  // Not using the default device vendor
513  if (!setup_default_.default_device_vendor()) {
514  for (std::size_t d = 0; d != dev_pool.size(); ++d) {
515  dev_pool[d].get_info(CL_DEVICE_VENDOR, name);
516  if (!setup_default_.check_device_vendor(name))
517  dev_select_idx[d] = false;
518  }
519  }
520 
521  // Not using the default device
522  if (!setup_default_.default_device()) {
523  for (std::size_t d = 0; d != dev_pool.size(); ++d) {
524  dev_pool[d].get_info(CL_DEVICE_NAME, name);
525  if (!setup_default_.check_device(name))
526  dev_select_idx[d] = false;
527  }
528  }
529 
530  // determine if we should remove intel GPU
531  std::size_t num = 0;
532  for (std::size_t d = 0; d != dev_pool.size(); ++d)
533  if (dev_select_idx[d] && dev_select_idx_igpu[d])
534  ++num;
535  if (num > 0) {
536  for (std::size_t d = 0; d != dev_pool.size(); ++d)
537  if (!dev_select_idx_igpu[d])
538  dev_select_idx[d] = false;
539  }
540 
541  std::vector<CLDevice> dev_select;
542  for (std::size_t d = 0; d != dev_pool.size(); ++d)
543  if (dev_select_idx[d])
544  dev_select.push_back(dev_pool[d]);
545 
546  return dev_select;
547  }
