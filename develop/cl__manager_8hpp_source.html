485  void setup_cl_manager(::cl_device_type dev_type)
486  {
487  setup_ = false;
488  ::cl_int status = CL_SUCCESS;
489 
490  bool setup_platform = platform_filter(dev_type);
491  if (!setup_platform) {
492  VSMC_RUNTIME_WARNING_OPENCL_CL_MANAGER_SETUP_PLATFORM;
493  return;
494  }
495 
496  ::cl_uint num_dev = 0;
497  status =
498  ::clGetDeviceIDs(platform_.get(), dev_type, 0, nullptr, &num_dev);
499  if (status != CL_SUCCESS || num_dev == 0) {
500  VSMC_RUNTIME_WARNING_OPENCL_CL_MANAGER_SETUP_CONTEXT;
501  VSMC_RUNTIME_WARNING_OPENCL_CL_MANAGER_SETUP_DEVICE;
502  return;
503  }
504 
505  std::vector<::cl_device_id> dev_pool_ptr(num_dev);
506  status = ::clGetDeviceIDs(
507  platform_.get(), dev_type, num_dev, dev_pool_ptr.data(), nullptr);
508  if (status != CL_SUCCESS) {
509  VSMC_RUNTIME_WARNING_OPENCL_CL_MANAGER_SETUP_CONTEXT;
510  VSMC_RUNTIME_WARNING_OPENCL_CL_MANAGER_SETUP_DEVICE;
511  return;
512  }
513 
514  std::vector<std::shared_ptr<CLDevice>> dev_pool;
515  for (auto ptr : dev_pool_ptr)
516  dev_pool.push_back(cl_device_make_shared(ptr));
517  std::vector<std::shared_ptr<CLDevice>> dev_select;
518  device_filter(dev_pool, dev_select);
519  if (dev_select.size() == 0) {
520  VSMC_RUNTIME_WARNING_OPENCL_CL_MANAGER_SETUP_CONTEXT;
521  VSMC_RUNTIME_WARNING_OPENCL_CL_MANAGER_SETUP_DEVICE;
522  return;
523  }
524 
525  std::vector<::cl_device_id> dev_select_ptr;
526  for (const auto &dev : dev_select)
527  dev_select_ptr.push_back(dev.get());
528  ::cl_context_properties context_properties[] = {CL_CONTEXT_PLATFORM,
529  reinterpret_cast<::cl_context_properties>(platform_.get()), 0};
530  ::cl_context ctx = ::clCreateContext(context_properties,
531  static_cast<::cl_uint>(dev_select_ptr.size()),
532  dev_select_ptr.data(), nullptr, nullptr, &status);
533  if (status != CL_SUCCESS) {
534  VSMC_RUNTIME_WARNING_OPENCL_CL_MANAGER_SETUP_CONTEXT;
535  VSMC_RUNTIME_WARNING_OPENCL_CL_MANAGER_SETUP_DEVICE;
536  return;
537  }
538  context_ = cl_context_make_shared(ctx);
539 
540  bool setup_device = set_device_vec();
541  if (!setup_device) {
542  VSMC_RUNTIME_WARNING_OPENCL_CL_MANAGER_SETUP_DEVICE;
543  return;
544  }
545  device_ = device_vec_.front();
546 
547  ::cl_command_queue cmd =
548  ::clCreateCommandQueue(context_.get(), device_.get(), 0, &status);
549  if (status != CL_SUCCESS) {
550  VSMC_RUNTIME_WARNING_OPENCL_CL_MANAGER_SETUP_COMMAND_QUEUE;
551  return;
552  }
553  command_queue_ = cl_command_queue_make_shared(cmd);
554 
555  check_opencl_version();
556 
557  setup_ = true;
558  }

560  bool platform_filter(::cl_device_type dev_type)
561  {
562  ::cl_int status = CL_SUCCESS;
563 
564  ::cl_uint num_plat = 0;
565  status = ::clGetPlatformIDs(0, nullptr, &num_plat);
566  if (status != CL_SUCCESS)
567  return false;
568 
569  std::vector<::cl_platform_id> plat_vec_ptr(num_plat);
570  status = ::clGetPlatformIDs(num_plat, plat_vec_ptr.data(), nullptr);
571  if (status != CL_SUCCESS)
572  return false;
573 
574  std::vector<std::shared_ptr<CLPlatform>> plat_vec;
575  for (auto ptr : plat_vec_ptr)
576  plat_vec.push_back(cl_platform_make_shared(ptr));
577 
578  // If not using default platform
579  if (!setup_default_.default_platform()) {
580  for (const auto &plat : plat_vec) {
581  std::size_t name_size = 0;
582  status = ::clGetPlatformInfo(
583  plat.get(), CL_PLATFORM_NAME, 0, nullptr, &name_size);
584  if (status != CL_SUCCESS)
585  continue;
586  std::vector<char> name_vec(name_size);
587  status = ::clGetPlatformInfo(plat.get(), CL_PLATFORM_NAME,
588  name_size, name_vec.data(), nullptr);
589  if (status != CL_SUCCESS)
590  continue;
591  name_vec.push_back(0);
592  std::string name(static_cast<const char *>(name_vec.data()));
593  if (setup_default_.check_platform(name)) {
594  platform_ = plat;
595  return true;
596  }
597  }
598 
599  return false;
600  }
601 
602  // Using default platform: finding the first that has the device type
603  for (const auto &plat : plat_vec) {
604  ::cl_uint num_dev = 0;
605  status = ::clGetDeviceIDs(
606  platform_.get(), dev_type, 0, nullptr, &num_dev);
607  if (status != CL_SUCCESS)
608  continue;
609  std::vector<::cl_device_id> dev_pool_ptr(num_dev);
610  status = ::clGetDeviceIDs(platform_.get(), dev_type, num_dev,
611  dev_pool_ptr.data(), nullptr);
612  if (status != CL_SUCCESS)
613  continue;
614  std::vector<std::shared_ptr<CLDevice>> dev_pool;
615  for (auto ptr : dev_pool_ptr)
616  dev_pool.push_back(cl_device_make_shared(ptr));
617  std::vector<std::shared_ptr<CLDevice>> dev_select;
618  device_filter(dev_pool, dev_select);
619  if (dev_select.size() != 0) {
620  platform_ = plat;
621  return true;
622  }
623  }
624 
625  return false;
626  }

640  if (status != CL_SUCCESS) {

645  status = ::clGetDeviceInfo(dev_pool[d].get(), CL_DEVICE_VENDOR,
646  name_size, name_vec.data(), nullptr);
647  if (status != CL_SUCCESS) {
648  dev_select_idx[d] = false;
649  continue;
650  }
651  name_vec.push_back(0);
652  std::string name(static_cast<const char *>(name_vec.data()));
653  if (!setup_default_.check_device_vendor(name))
654  dev_select_idx[d] = false;
655  }
656  }
657 
658  // Not using the default device
659  if (!setup_default_.default_device()) {
660  for (std::size_t d = 0; d != dev_pool.size(); ++d) {
661  std::size_t name_size = 0;
662  status = ::clGetDeviceInfo(

664  if (status != CL_SUCCESS) {
665  dev_select_idx[d] = false;
666  continue;
667  }
668  std::vector<char> name_vec(name_size);
669  status = ::clGetDeviceInfo(dev_pool[d].get(), CL_DEVICE_NAME,
670  name_size, name_vec.data(), nullptr);
671  if (status != CL_SUCCESS) {
672  dev_select_idx[d] = false;
673  continue;
674  }
675  name_vec.push_back(0);
676  std::string name(static_cast<const char *>(name_vec.data()));
677  if (!setup_default_.check_device(name))

679  }

683  if (dev_select_idx[d])
684  dev_select.push_back(dev_pool[d]);
