405  template <typename Func>
406  typename cxx11::enable_if<
407  !cxx11::is_same<Func, std::size_t>::value &&
408  !cxx11::is_convertible<Func, std::size_t>::value, std::size_t>::type
409  profile_kernel (::cl::Kernel &kern, std::size_t N,
410  const Func &func, std::size_t lmin = 0, std::size_t repeat = 10)
411  {
412  cl::size_t<3> reqd_size;
413  try {
414  kern.getWorkGroupInfo(device_,
415  CL_KERNEL_COMPILE_WORK_GROUP_SIZE, &reqd_size);
416  } catch (::cl::Error) {
417  reqd_size[0] = 0;
418  }
419 
420  if (reqd_size[0] != 0)
421  return reqd_size[0];
422 
423  std::size_t factor;
424  std::size_t lmax;
425  std::size_t mmax;
426  cl_minmax_local_size(kern, device_, factor, lmax, mmax);
427  if (lmax == 0)
428  return 0;
429 
430  if (lmin != 0 && lmin <= lmax) {
431  factor = lmin;
432  mmax = lmax / factor;
433  lmax = mmax * factor;
434  }
435 
436  double time = std::numeric_limits<double>::max VSMC_MNE ();
437  std::size_t lsize = lmax;
438  vsmc::StopWatch watch;
439  Func f(func);
440  for (std::size_t m = mmax; m >= 1; --m) {
441  std::size_t l = m * factor;
442  f(kern);
443  run_kernel(kern, N, l);
444  watch.reset();
445  watch.start();
446  for (std::size_t r = 0; r != repeat; ++r) {
447  f(kern);
448  run_kernel(kern, N, l);
449  }
450  watch.stop();
451  if (time > watch.milliseconds()) {
452  time = watch.milliseconds();
453  lsize = l;
454  }
455  }
456 
457  return lsize;
458  }

485  void setup_cl_manager (cl_device_type dev_type)
486  {
487  setup_ = false;
488 
489  bool setup_platform = platform_filter(dev_type);
490  VSMC_RUNTIME_WARNING_CL_MANAGER_SETUP_PLATFORM;
491  if (!setup_platform) return;
492 
493  bool setup_context = false;
494  bool setup_device = false;
495  try {
496  std::vector< ::cl::Device> dev_pool;
497  std::vector< ::cl::Device> dev_select;
498  platform_.getDevices(dev_type, &dev_pool);
499  device_filter(dev_pool, dev_select);
500  if (dev_select.size() != 0) {
501  cl_context_properties context_properties[] = {
502  static_cast<cl_context_properties>(CL_CONTEXT_PLATFORM),
503  reinterpret_cast<cl_context_properties>(platform_()), 0
504  };
505  context_ = ::cl::Context(dev_select, context_properties);
506  setup_context = true;
507  device_vec_ = context_.getInfo<CL_CONTEXT_DEVICES>();
508  device_ = device_vec_[0];
509  setup_device = true;
510  }
511  } catch (::cl::Error) {}
512  VSMC_RUNTIME_WARNING_CL_MANAGER_SETUP_CONTEXT;
513  VSMC_RUNTIME_WARNING_CL_MANAGER_SETUP_DEVICE;
514  if (!setup_context) return;
515  if (!setup_device) return;
516 
517  bool setup_command_queue = false;
518  try {
519  command_queue_ = ::cl::CommandQueue(context_, device_, 0);
520  setup_command_queue = true;
521  } catch (::cl::Error) {}
522  VSMC_RUNTIME_WARNING_CL_MANAGER_SETUP_COMMAND_QUEUE;
523  if (!setup_command_queue) return;
524 
525  setup_ = true;
526  }

528  bool platform_filter (cl_device_type dev_type)
529  {
530  std::vector< ::cl::Platform> platform_vec;
531  try {
532  ::cl::Platform::get(&platform_vec);
533  } catch (::cl::Error) {
534  platform_vec.clear();
535  }
536  if (platform_vec.size() == 0)
537  return false;
538 
539  // If not using default platform
540  if (!setup_default_.default_platform()) {
541  for (std::size_t p = 0; p != platform_vec.size(); ++p) {
542  try {
543  std::string name;
544  platform_vec[p].getInfo(CL_PLATFORM_NAME, &name);
545  if (setup_default_.check_platform(name)) {
546  platform_ = platform_vec[p];
547  return true;
548  }
549  } catch (::cl::Error) {}
550  }
551  }
552 
553  // Using default platform: finding the first that has the device type
554  for (std::size_t p = 0; p != platform_vec.size(); ++p) {
555  try {
556  std::vector< ::cl::Device> dev_pool;
557  std::vector< ::cl::Device> dev_select;
558  platform_vec[p].getDevices(dev_type, &dev_pool);
559  device_filter(dev_pool, dev_select);
560  if (dev_select.size() != 0) {
561  platform_ = platform_vec[p];
562  return true;
563  }
564  } catch (::cl::Error) {}
565  }
566 
567  return false;
568  }
