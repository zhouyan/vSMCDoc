306  static void increment_block_nskip(ctr_type &ctr, T nskip, std::false_type)
307  {
308  T m = ctr.back() & mask_lo_;
309  T b = ctr.back();
310  b <<= 8;
311  b >>= 8;
312 
313  if (nskip <= mask_hi_ - b) {
314  b += nskip;
315  b &= mask_hi_;
316  b ^= m;
317  ctr.back() = b;
318  return;
319  }

322  while (nskip > mask_hi_)
323  nskip -= mask_hi_;
324  b = nskip;
325  b &= mask_hi_;
326  b ^= m;
327  ctr.back() = b;
