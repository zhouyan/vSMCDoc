657 template <SIMD ISA>
658 VSMC_STRONG_INLINE void *memset_simd (
659  void *dst, int ch, std::size_t n)
660 {
661  if (n < traits::SIMDTrait<ISA>::alignment) {
662  memset_0<ISA>(dst, ch, n);
663  return dst;
664  }
665 
666  unsigned flag = cstring_is_aligned<ISA>(dst);
667  if (n <= traits::SIMDTrait<ISA>::alignment *
668  traits::SIMDTrait<ISA>::grainsize) {
669  memset_n_switch<ISA, traits::SIMDTrait<ISA>::grainsize>(
670  dst, ch, n, flag);
671  return dst;
672  }
673 
674  char *dstc = static_cast<char *>(dst);
675  std::size_t offset = reinterpret_cast<uintptr_t>(dstc) % 64;
676  if (offset != 0) {
677  offset = 64 - offset;
678  memset_n_switch<ISA, 32 / traits::SIMDTrait<ISA>::alignment>(
679  dstc, ch, offset, flag);
680  n -= offset;
681  dstc += offset;
682  }
683  flag = CStringNonTemporalThreshold::instance().over(n) | 2;
684  memset_l_switch<ISA, traits::SIMDTrait<ISA>::grainsize>(dstc, ch, n, flag);
685 
686  return dst;
687 }

689 template <SIMD ISA>
690 VSMC_STRONG_INLINE void *memcpy_simd (
691  void *dst, const void *src, std::size_t n)
692 {
693  if (dst == src)
694  return dst;
695 
696  if (n < traits::SIMDTrait<ISA>::alignment) {
697  memcpy_0<ISA>(dst, src, n);
698  return dst;
699  }
700 
701  unsigned flag = cstring_is_aligned<ISA>(dst);
702  flag |= cstring_is_aligned<ISA>(src) << 1;
703  if (n <= traits::SIMDTrait<ISA>::alignment *
704  traits::SIMDTrait<ISA>::grainsize) {
705  memcpy_n_switch<ISA, traits::SIMDTrait<ISA>::grainsize>(
706  dst, src, n, flag);
707  return dst;
708  }
709 
710  char *dstc = static_cast<char *>(dst);
711  const char *srcc = static_cast<const char *>(src);
712  std::size_t offset = reinterpret_cast<uintptr_t>(dstc) % 64;
713  if (offset != 0) {
714  offset = 64 - offset;
715  memcpy_n_switch<ISA, 32 / traits::SIMDTrait<ISA>::alignment>(
716  dstc, srcc, offset, flag);
717  n -= offset;
718  dstc += offset;
719  srcc += offset;
720  }
721  flag = CStringNonTemporalThreshold::instance().over(n) | 2;
722  flag |= cstring_is_aligned<ISA>(srcc) << 1;
723  memcpy_l_switch<ISA, traits::SIMDTrait<ISA>::grainsize>(
724  dstc, srcc, n, flag);
725 
726  return dst;
727 }
