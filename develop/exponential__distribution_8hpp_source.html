182  RNGType &rng, std::size_t n, RealType *r, RealType lambda)
183 {
184  const std::size_t k = 1000;
185  const std::size_t m = n / k;
186  const std::size_t l = n % k;
187  for (std::size_t i = 0; i != m; ++i)
188  internal::exponential_distribution_impl(rng, k, r + i * k, lambda);
189  internal::exponential_distribution_impl(rng, l, r + m * k, lambda);
190 }
