62 template <typename RealType, typename RNGType>
63 inline void exponential_distribution(
64  RNGType &rng, std::size_t n, RealType *r, RealType lambda = 1)
65 {
66  const std::size_t k = 1000;
67  const std::size_t m = n / k;
68  const std::size_t l = n % k;
69  for (std::size_t i = 0; i != m; ++i)
70  internal::exponential_distribution_impl(rng, k, r + i * k, lambda);
71  internal::exponential_distribution_impl(rng, l, r + m * k, lambda);
72 }
