92 template <typename RealType, typename RNGType>
93 inline void exponential_distribution(
94  RNGType &rng, std::size_t n, RealType *r, RealType lambda)
95 {
96  const std::size_t k = 1000;
97  const std::size_t m = n / k;
98  const std::size_t l = n % k;
99  for (std::size_t i = 0; i != m; ++i)
100  internal::exponential_distribution_impl(rng, k, r + i * k, lambda);
101  internal::exponential_distribution_impl(rng, l, r + m * k, lambda);
102 }
