195 template <typename RealType, typename RNGType>
196 inline void extreme_value_distribution(
197  RNGType &rng, std::size_t n, RealType *r, RealType a, RealType b)
198 {
199  const std::size_t k = 1000;
200  const std::size_t m = n / k;
201  const std::size_t l = n % k;
202  for (std::size_t i = 0; i != m; ++i)
203  internal::extreme_value_distribution_impl(rng, k, r + i * k, a, b);
204  internal::extreme_value_distribution_impl(rng, l, r + m * k, a, b);
205 }
