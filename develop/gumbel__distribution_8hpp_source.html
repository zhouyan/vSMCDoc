63 template <typename RealType, typename RNGType>
64 inline void gumbel_distribution(RNGType &rng, std::size_t n, RealType *r,
65  RealType location = 0, RealType scale = 1)
66 {
67  const std::size_t k = 1000;
68  const std::size_t m = n / k;
69  const std::size_t l = n % k;
70  for (std::size_t i = 0; i != m; ++i)
71  internal::gumbel_distribution_impl(rng, k, r + i * k, location, scale);
72  internal::gumbel_distribution_impl(rng, l, r + m * k, location, scale);
73 }
