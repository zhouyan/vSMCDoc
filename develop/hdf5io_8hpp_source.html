242 inline ::hsize_t hdf5size(
243  const std::string &file_name, const std::string &data_name)
244 {
245  ::hid_t datafile =
246  ::H5Fopen(file_name.c_str(), H5F_ACC_RDONLY, H5P_DEFAULT);
247  ::hid_t dataset = ::H5Dopen(datafile, data_name.c_str(), H5P_DEFAULT);
248  ::hid_t dataspace = ::H5Dget_space(dataset);
249  ::hid_t datatype = ::H5Dget_type(dataset);
250  std::size_t n =
251  static_cast<std::size_t>(H5Sget_simple_extent_npoints(dataspace));
252  std::size_t b = H5Tget_size(datatype);
253  std::size_t bytes = n * b;
254 
255  ::H5Tclose(datatype);
256  ::H5Sclose(dataspace);
257  ::H5Dclose(dataset);
258  ::H5Fclose(datafile);
259 
260  return bytes;
261 }

275 inline OutputIter hdf5load(const std::string &file_name,
276  const std::string &data_name, OutputIter first)
277 {
278  std::size_t n = hdf5size(file_name, data_name) / sizeof(T);
279  internal::HDF5LoadDataPtr<T> data_ptr;
280  data_ptr.set(n, first);
281  T *data = data_ptr.get();
282 
283  ::hid_t datafile =
284  ::H5Fopen(file_name.c_str(), H5F_ACC_RDONLY, H5P_DEFAULT);
285  ::hid_t dataset = ::H5Dopen(datafile, data_name.c_str(), H5P_DEFAULT);
286  ::hid_t datatype = ::H5Dget_type(dataset);
287  ::H5Dread(dataset, datatype, H5S_ALL, H5S_ALL, H5P_DEFAULT, data);
288  if (data_ptr.is_raw_ptr()) {
289  first += n;
290  } else {
291  for (std::size_t i = 0; i != n; ++i, ++first)
292  *first = data[i];
293  }
294 
295  ::H5Tclose(datatype);
296  ::H5Dclose(dataset);
297  ::H5Fclose(datafile);
298 
299  return first;
300 }

695 inline void hdf5store(const Sampler<T> &sampler, const std::string &file_name,
696  const std::string &data_name, bool append)
697 {

706

708  Vector<std::string> header_int(ncol_int);
709  Vector<size_type> data_int(nrow * ncol_int);
710  sampler.summary_header_int(header_int.begin());

712  bool use_int = internal::hdf5store_int(
713  data_int.size(), data_int.data(), std::is_signed<size_type>());
714  if (use_int) {
715  Vector<int> data_int_small(data_int.size());
716  std::copy(data_int.begin(), data_int.end(), data_int_small.begin());
717  const int *data_ptr_int = data_int_small.data();
718  for (std::size_t j = 0; j != ncol_int; ++j, data_ptr_int += nrow) {
719  hdf5store_list_insert<int>(
720  nrow, file_name, data_name, data_ptr_int, header_int[j]);
721  }
722  } else {
723  const size_type *data_ptr_int = data_int.data();
724  for (std::size_t j = 0; j != ncol_int; ++j, data_ptr_int += nrow) {
725  hdf5store_list_insert<size_type>(

727  }
728  }
729

739  }
