211 inline ::hsize_t hdf5size (const std::string &file_name,
212  const std::string &data_name)
213 {
214  ::hid_t datafile = ::H5Fopen(file_name.c_str(),
215  H5F_ACC_RDONLY, H5P_DEFAULT);
216  ::hid_t dataset = ::H5Dopen(datafile, data_name.c_str(), H5P_DEFAULT);
217  ::hid_t dataspace = ::H5Dget_space(dataset);
218  ::hid_t datatype = ::H5Dget_type(dataset);
219  std::size_t n = static_cast<std::size_t>(H5Sget_simple_extent_npoints(
220  dataspace));
221  std::size_t b = H5Tget_size(datatype);
222  std::size_t bytes = n * b;
223 
224  ::H5Tclose(datatype);
225  ::H5Sclose(dataspace);
226  ::H5Dclose(dataset);
227  ::H5Fclose(datafile);
228 
229  return bytes;
230 }

234 template <typename T, typename OutputIter>
235 inline OutputIter hdf5load (const std::string &file_name,
236  const std::string &data_name, OutputIter first)
237 {
238  std::size_t n = hdf5size(file_name, data_name) / sizeof(T);
239  internal::HDF5LoadDataPtr<T> data_ptr;
240  data_ptr.set(n, first);
241  T *data = data_ptr.get();
242 
243  ::hid_t datafile = ::H5Fopen(file_name.c_str(),
244  H5F_ACC_RDONLY, H5P_DEFAULT);
245  ::hid_t dataset = ::H5Dopen(datafile, data_name.c_str(), H5P_DEFAULT);
246  ::hid_t datatype = ::H5Dget_type(dataset);
247  ::H5Dread(dataset, datatype, H5S_ALL, H5S_ALL, H5P_DEFAULT, data);
248  if (data_ptr.set_ptr()) {
249  first += n;
250  } else {
251  for (std::size_t i = 0; i != n; ++i, ++first)
252  *first = data[i];
253  }
254 
255  ::H5Tclose(datatype);
256  ::H5Dclose(dataset);
257  ::H5Fclose(datafile);
258 
259  return first;
260 }

609 template <typename T>
610 inline void hdf5store (const Sampler<T> &sampler,
611  const std::string &file_name, const std::string &data_name,
612  bool append = false)
613 {
614  std::size_t nrow = sampler.iter_size();
615 
616  if (nrow == 0)
617  return;
618 
619  std::size_t ncol_int = sampler.summary_header_size_int();
620  std::vector<std::string> header_int(ncol_int);
621  std::vector<int> data_int(nrow * ncol_int);
622  sampler.summary_header_int(header_int.begin());
623  sampler.template summary_data_int<ColMajor>(data_int.begin());
624  std::vector<const int *> data_ptr_int(ncol_int);
625  for (std::size_t j = 0; j != ncol_int; ++j)
626  data_ptr_int[j] = &data_int[0] + j * nrow;
627 
628  std::size_t ncol = sampler.summary_header_size();
629  std::vector<std::string> header(ncol);
630  std::vector<double> data(nrow * ncol);
631  sampler.summary_header(header.begin());
632  sampler.template summary_data<ColMajor>(data.begin());
633  std::vector<const double *> data_ptr(ncol);
634  for (std::size_t j = 0; j != ncol; ++j)
635  data_ptr[j] = &data[0] + j * nrow;
636 
637  hdf5store_list_empty(file_name, data_name, append);
638 
639  for (std::size_t j = 0; j != ncol_int; ++j)
640  hdf5store_list_insert<int>(nrow, file_name, data_name,
641  data_ptr_int[j], header_int[j]);
642 
643  for (std::size_t j = 0; j != ncol; ++j)
644  hdf5store_list_insert<double>(nrow, file_name, data_name,
645  data_ptr[j], header[j]);
646 }
