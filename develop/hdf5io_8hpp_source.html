244 inline ::hsize_t hdf5size(
245  const std::string &file_name, const std::string &data_name)
246 {
247  ::hid_t datafile =
248  ::H5Fopen(file_name.c_str(), H5F_ACC_RDONLY, H5P_DEFAULT);
249  ::hid_t dataset = ::H5Dopen(datafile, data_name.c_str(), H5P_DEFAULT);
250  ::hid_t dataspace = ::H5Dget_space(dataset);
251  ::hid_t datatype = ::H5Dget_type(dataset);
252  std::size_t n =
253  static_cast<std::size_t>(H5Sget_simple_extent_npoints(dataspace));
254  std::size_t b = H5Tget_size(datatype);
255  std::size_t bytes = n * b;
256 
257  ::H5Tclose(datatype);
258  ::H5Sclose(dataspace);
259  ::H5Dclose(dataset);
260  ::H5Fclose(datafile);
261 
262  return bytes;
263 }

267 template <typename T, typename OutputIter>
268 inline OutputIter hdf5load(const std::string &file_name,
269  const std::string &data_name, OutputIter first)
270 {
271  std::size_t n = hdf5size(file_name, data_name) / sizeof(T);
272  internal::HDF5LoadDataPtr<T> data_ptr;
273  data_ptr.set(n, first);
274  T *data = data_ptr.get();
275 
276  ::hid_t datafile =
277  ::H5Fopen(file_name.c_str(), H5F_ACC_RDONLY, H5P_DEFAULT);
278  ::hid_t dataset = ::H5Dopen(datafile, data_name.c_str(), H5P_DEFAULT);
279  ::hid_t datatype = ::H5Dget_type(dataset);
280  ::H5Dread(dataset, datatype, H5S_ALL, H5S_ALL, H5P_DEFAULT, data);
281  if (data_ptr.set_ptr()) {
282  first += n;
283  } else {
284  for (std::size_t i = 0; i != n; ++i, ++first)
285  *first = data[i];
286  }
287 
288  ::H5Tclose(datatype);
289  ::H5Dclose(dataset);
290  ::H5Fclose(datafile);
291 
292  return first;
293 }

646 template <typename T>
647 inline void hdf5store(const Sampler<T> &sampler, const std::string &file_name,
648  const std::string &data_name, bool append = false)
649 {
650  typedef typename Sampler<T>::size_type size_type;
651 
652  std::size_t nrow = sampler.iter_size();
653 
654  if (nrow == 0)
655  return;
656 
657  std::size_t ncol_int = sampler.summary_header_size_int();
658  std::vector<std::string> header_int(ncol_int);
659  std::vector<size_type> data_int(nrow * ncol_int);
660  sampler.summary_header_int(header_int.begin());
661  sampler.template summary_data_int<ColMajor>(data_int.begin());
662  std::vector<const size_type *> data_ptr_int(ncol_int);
663  for (std::size_t j = 0; j != ncol_int; ++j)
664  data_ptr_int[j] = data_int.data() + j * nrow;
665 
666  std::size_t ncol = sampler.summary_header_size();
667  std::vector<std::string> header(ncol);
668  std::vector<double> data(nrow * ncol);
669  sampler.summary_header(header.begin());
670  sampler.template summary_data<ColMajor>(data.begin());
671  std::vector<const double *> data_ptr(ncol);
672  for (std::size_t j = 0; j != ncol; ++j)
673  data_ptr[j] = data.data() + j * nrow;
674 
675  hdf5store_list_empty(file_name, data_name, append);
676 
677  for (std::size_t j = 0; j != ncol_int; ++j)
678  hdf5store_list_insert<size_type>(
679  nrow, file_name, data_name, data_ptr_int[j], header_int[j]);
680 
681  for (std::size_t j = 0; j != ncol; ++j)
682  hdf5store_list_insert<double>(
683  nrow, file_name, data_name, data_ptr[j], header[j]);
684 }
