88 template <std::size_t K, typename RealType, typename RNGType>
89 inline void laplace_distribution_impl(
90  RNGType &rng, std::size_t n, RealType *r, RealType a, RealType b)
91 {
92  Array<RealType, K> s;
93  u01_oo_distribution(rng, n, r);
94  sub(n, r, static_cast<RealType>(0.5), r);
95  for (std::size_t i = 0; i != n; ++i) {
96  if (r[i] > 0) {
97  r[i] = 1 - 2 * r[i];
98  s[i] = -b;
99  } else {
100  r[i] = 1 + 2 * r[i];
101  s[i] = b;
102  }
103  }
104  log(n, r, r);
105  fma(n, s.data(), r, a, r);
106 }
