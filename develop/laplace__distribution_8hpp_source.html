48 template <typename RealType, typename RNGType>
49 void laplace_distribution(RNGType &rng, std::size_t n, RealType *r,
50  RealType location = 0, RealType scale = 1)
51 {
52  U01DistributionType<RNGType, RealType> runif;
53  Vector<RealType> b(n);
54  for (std::size_t i = 0; i != n; ++i) {
55  r[i] = runif(rng) - 0.5;
56  if (r[i] > 0) {
57  r[i] *= 2;
58  b[i] = -scale;
59  } else {
60  r[i] *= -2;
61  b[i] = scale;
62  }
63  }
64  math::vLog1p(n, r, r);
65  math::vMul(n, r, b.data(), r);
66  for (std::size_t i = 0; i != n; ++i)
67  r[i] += location;
68 }
