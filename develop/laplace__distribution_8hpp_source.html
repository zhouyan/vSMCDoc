84 template <std::size_t K, typename RealType, typename RNGType>
85 inline void laplace_distribution_impl(
86  RNGType &rng, std::size_t n, RealType *r, RealType a, RealType b)
87 {
88  RealType s[K];
89  u01_oc_distribution(rng, n, r);
90  sub(n, r, static_cast<RealType>(0.5), r);
91  for (std::size_t i = 0; i != n; ++i) {
92  if (r[i] > 0) {
93  r[i] = 1 - 2 * r[i];
94  s[i] = -b;
95  } else {
96  r[i] = 1 + 2 * r[i];
97  s[i] = b;
98  }
99  }
100  log(n, r, r);
101  fma(n, s, r, a, r);
102 }

111 {
112  const std::size_t k = 1000;
113  const std::size_t m = n / k;
114  const std::size_t l = n % k;
115  for (std::size_t i = 0; i != m; ++i)
116  internal::laplace_distribution_impl<k>(rng, k, r + i * k, a, b);
117  internal::laplace_distribution_impl<k>(rng, l, r + m * k, a, b);
118 }
