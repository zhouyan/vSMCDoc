190 template <typename RealType, typename RNGType>
191 inline void levy_distribution(
192  RNGType &rng, std::size_t n, RealType *r, RealType a, RealType b)
193 {
194  const std::size_t k = 1000;
195  const std::size_t m = n / k;
196  const std::size_t l = n % k;
197  for (std::size_t i = 0; i != m; ++i)
198  internal::levy_distribution_impl<k>(rng, k, r + i * k, a, b);
199  internal::levy_distribution_impl<k>(rng, l, r + m * k, a, b);
200 }
