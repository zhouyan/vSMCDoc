97 inline void lognormal_distribution(RNGType &rng, std::size_t n, RealType *r,
98  RealType logmean, RealType logstddev)
99 {
100  const std::size_t k = 1000;
101  const std::size_t m = n / k;
102  const std::size_t l = n % k;
103  for (std::size_t i = 0; i != m; ++i, r += k)
104  internal::lognormal_distribution_impl(rng, k, r, logmean, logstddev);
105  internal::lognormal_distribution_impl(rng, l, r, logmean, logstddev);
106 }
