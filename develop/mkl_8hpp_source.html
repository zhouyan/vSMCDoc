651 template <typename CharT, typename Traits>
652 inline std::basic_ostream<CharT, Traits> &operator<<(
653  std::basic_ostream<CharT, Traits> &os, const MKLStream &stream)
654 {
655  if (!os)
656  return os;
657 
658  std::size_t n = static_cast<std::size_t>(stream.get_size());
659  std::size_t m = sizeof(std::uintmax_t);
660  if (n % m != 0)
661  n += m - n % m;
662  n /= m;
663  Vector<std::uintmax_t> s(n);
664  stream.save_m(reinterpret_cast<char *>(s.data()));
665 
666  os << stream.get_brng() << ' ';
667  os << s;
668 
669  return os;
670 }

895  r += remain;
896  n -= remain;

902  stream_, static_cast<MKL_INT>(m), r);

904 
905  generate();

907  index_ = l;
908  }

919  void discard(long long nskip)
920  {
921  VSMC_RUNTIME_ASSERT_RNG_MKL_DISCARD;
922 
923  if (nskip == 0)
924  return;
925 
926  const long long remain = static_cast<long long>(M_ - index_);
927  if (nskip <= remain) {
928  index_ += static_cast<std::size_t>(nskip);
929  return;
930  }
931  nskip -= remain;
932  index_ = M_;
933 
934  ::VSLBRngProperties properties;
935  MKLStream::get_brng_properties(stream_.get_brng(), &properties);
936  int bits = properties.NBits;
937  long long M = static_cast<long long>(M_);
938  long long m = nskip / M * M;
939  if (Bits >= bits)
940  m *= Bits / bits + (Bits % bits == 0 ? 0 : 1);
941  switch (stream_.get_brng()) {
942  case VSL_BRNG_MCG31: stream_.skip_ahead(m); break;
943  case VSL_BRNG_MRG32K3A: stream_.skip_ahead(m); break;
944  case VSL_BRNG_MCG59: stream_.skip_ahead(m); break;
945  case VSL_BRNG_WH: stream_.skip_ahead(m); break;
946  case VSL_BRNG_MT19937: stream_.skip_ahead(m); break;
947  case VSL_BRNG_SFMT19937: stream_.skip_ahead(m); break;
948  case VSL_BRNG_SOBOL: stream_.skip_ahead(m); break;
949  case VSL_BRNG_NIEDERR: stream_.skip_ahead(m); break;
950 #if INTEL_MKL_VERSION >= 110300
951  case VSL_BRNG_PHILOX4X32X10: stream_.skip_ahead(m); break;
952  case VSL_BRNG_ARS5: stream_.skip_ahead(m); break;
953 #endif
954  default:
955  if (MKLStream::has_skip_ahead(stream_.get_brng())) {
956  stream_.skip_ahead(m);
957  break;
958  }
959  while (nskip > M) {
960  generate();
961  nskip -= M;
962  }
963  };
964  generate();
965  index_ = static_cast<std::size_t>(nskip % M);
966  }
