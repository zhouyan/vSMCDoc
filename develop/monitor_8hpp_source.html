312  buffer_.resize(N * dim_);

314  math::gemv(ColMajor, NoTrans, dim_, N, 1, buffer_.data(), dim_,
315  particle.weight_set().weight_data(), 1, 0, result_.data(), 1);
316  push_back(iter);
317  }
