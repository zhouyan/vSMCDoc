295  void eval(
296  std::size_t iter, const Particle<T> &particle, MonitorStage stage)
297  {
298  if (!recording_)
299  return;
300 
301  if (stage != stage_)
302  return;
303 
304  VSMC_RUNTIME_ASSERT_CORE_MONITOR_FUNCTOR(eval_, eval, EVALUATION);
305 
306  result_.resize(dim_);
307  if (record_only_) {
308  eval_(iter, dim_, particle, result_.data());
309  push_back(iter);
310 
311  return;
312  }
313 
314  const std::size_t N = static_cast<std::size_t>(particle.size());
315  buffer_.resize(N * dim_);
316  eval_(iter, dim_, particle, buffer_.data());
317 #if VSMC_HAS_MKL
318  if (ss_task_.ptr() == nullptr) {
319  MKL_INT p = static_cast<MKL_INT>(dim_);
320  MKL_INT n = static_cast<MKL_INT>(N);
321  MKL_INT xstorage = VSL_SS_MATRIX_STORAGE_COLS;
322  ss_task_.reset(
323  &p, &n, &xstorage, buffer_.data(), result_.data(), nullptr);
324  }
325  ::vsldSSCompute(ss_task_.ptr(), VSL_SS_MEAN, VSL_SS_METHOD_FAST);
326 #ifdef VSMC_CBLAS_INT
327  ::cblas_dgemv(::CblasColMajor, ::CblasNoTrans,
328  static_cast<VSMC_CBLAS_INT>(dim_), static_cast<VSMC_CBLAS_INT>(N),
329  1, buffer_.data(), static_cast<VSMC_CBLAS_INT>(dim_),
330  particle.weight_set().weight_data(), 1, 0, result_.data(), 1);
331 #else // VSMC_CBLAS_INT
332  const double *wptr = particle.weight_set().weight_data();
333  const double *bptr = buffer_.data();
334  std::fill(result_.begin(), result_.end(), 0.0);
335  for (std::size_t i = 0; i != N; ++i)
336  for (std::size_t d = 0; d != dim_; ++d, ++bptr)
337  result_[d] += particle.weight_set().weight(i) * (*bptr);
338 #endif // VSMC_CBLAS_INT
339 #endif // VSMC_HAS_MKL
340  push_back(iter);
341  }
