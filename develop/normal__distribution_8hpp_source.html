181  template <typename RNGType>
182  result_type operator()(RNGType &rng)
183  {
184  if (!param_.saved_) {
185  UniformRealDistributionType<RNGType, RealType> runif(-1, 1);
186  result_type u = 0;
187  result_type v = 0;
188  result_type s = 0;
189  do {
190  u = runif(rng);
191  v = runif(rng);
192  s = u * u + v * v;
193  } while (s > 1 || s <= 0);
194  s = std::sqrt(-2 * std::log(s) / s);
195  param_.u_ = u * s;
196  param_.v_ = v * s;
197  }
198  param_.saved_ = !param_.saved_;
199  double z = param_.saved_ ? param_.u_ : param_.v_;
200 
201  return param_.mean_ + param_.stddev_ * z;
202  }
