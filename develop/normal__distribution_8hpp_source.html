174  template <typename RNGType>
175  result_type operator()(RNGType &rng)
176  {
177  if (param_.saved_) {
178  param_.saved_ = false;
179  return param_.v_;
180  }
181 
182  U01OCDistribution<RealType> runif;
183  result_type u = runif(rng);
184  result_type v = runif(rng);
185  result_type s = param_.stddev_ * std::sqrt(-2 * std::log(u));
186  v *= const_pi_2<result_type>();
187  u = std::cos(v);
188  v = std::sin(v);
189  param_.v_ = param_.mean_ + v * s;
190  param_.saved_ = true;
191 
192  return param_.mean_ + u * s;
193  }

210 template <std::size_t K, typename RealType, typename RNGType>
211 inline void normal_distribution_impl(
212  RNGType &rng, std::size_t n, RealType *r, RealType mean, RealType stddev)
213 {
214  RealType s[K / 2];
215  const std::size_t nu = n / 2;
216  RealType *const u1 = r;
217  RealType *const u2 = r + nu;
218  u01_oc_distribution(rng, n, r);
219  log(nu, u1, s);
220  mul(nu, static_cast<RealType>(-2), s, s);
221  sqrt(nu, s, s);
222  mul(nu, const_pi_2<RealType>(), u2, u2);
223  sincos(nu, u2, u1, u2);
224  mul(nu, stddev, s, s);
225  fma(nu, mean, s, u1, u1);
226  fma(nu, mean, s, u2, u2);
227 }

254 template <typename RealType, typename RNGType>
255 inline void normal_distribution(
256  RNGType &rng, std::size_t n, RealType *r, RealType mean, RealType stddev)
257 {
258  const std::size_t k = 1000;
259  const std::size_t m = n / k;
260  const std::size_t l = n % k;
261  for (std::size_t i = 0; i != m; ++i, r += k)
262  internal::normal_distribution_impl<k>(rng, k, r, mean, stddev);
263  internal::normal_distribution_impl<k>(rng, l, r, mean, stddev);
264  if (n % 2 != 0) {
265  U01OCDistribution<RealType> runif;
266  RealType u = runif(rng);
267  RealType v = runif(rng);
268  r[l - 1] = mean +
269  stddev * std::sqrt(-2 * std::log(u)) *
270  std::cos(const_pi_2<RealType>() * v);
271  }
272 }
