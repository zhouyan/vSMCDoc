50 template <typename RealType, typename RNGType>
51 inline void normal_distribution_impl(RNGType &rng, std::size_t n, RealType *r,
52  RealType mean, RealType stddev, RealType *s)
53 {
54  u01_distribution(rng, n, r);
55  const std::size_t nu = n / 2;
56  RealType *const u1 = r;
57  RealType *const u2 = r + nu;
58  math::vLn(nu, u1, s);
59  math::vMul(nu, static_cast<RealType>(-2), s, s);
60  math::vSqrt(nu, s, s);
61  math::vMul(nu, math::pi_2<RealType>(), u2, u2);
62  math::vSin(nu, u2, u1);
63  math::vCos(nu, u2, u2);
64  math::vMul(nu, u1, s, u1);
65  math::vMul(nu, u2, s, u2);
66  for (std::size_t i = 0; i != n; ++i)
67  r[i] = mean + stddev * r[i];
68 }

74 template <typename RealType, typename RNGType>
75 inline void normal_distribution(RNGType &rng, std::size_t n, RealType *r,
76  RealType mean = 0, RealType stddev = 1)
77 {
78  const std::size_t k = 1000;
79  const std::size_t m = n / k;
80  const std::size_t l = n % k;
81  RealType s[k / 2];
82  for (std::size_t i = 0; i != m; ++i)
83  internal::normal_distribution_impl(rng, k, r + i * k, mean, stddev, s);
84  internal::normal_distribution_impl(rng, l, r + m * k, mean, stddev, s);
85  if (n % 2 != 0) {
86  U01DistributionType<RNGType, RealType> runif;
87  RealType v1 = runif(rng);
88  RealType v2 = runif(rng);
89  r[n - 1] = mean +
90  stddev * std::sqrt(-2 * std::log(v1)) *
91  std::cos(math::pi_2<RealType>() * v2);
92  }
93 }
