199  template <typename RNGType>
200  result_type generate(RNGType &rng, const param_type &param)
201  {
202  result_type z = 0;
203  if (saved_) {
204  z = v_;
205  saved_ = false;
206  } else {
207  U01Distribution<RealType> u01;
208  result_type u1 = std::sqrt(-2 * std::log(u01(rng)));
209  result_type u2 = const_pi_2<result_type>() * u01(rng);
210  z = u1 * std::cos(u2);
211  v_ = u1 * std::sin(u2);
212  saved_ = true;
213  }
214 
215  return param.mean() + param.stddev() * z;

222 template <std::size_t K, typename RealType, typename RNGType>
223 inline void normal_distribution_impl(
224  RNGType &rng, std::size_t n, RealType *r, RealType mean, RealType stddev)
225 {
226  Array<RealType, K / 2> s;
227  const std::size_t nu = n / 2;
228  RealType *const u1 = r;
229  RealType *const u2 = r + nu;
230  u01_distribution(rng, n, r);
231  log(nu, u1, s.data());
232  mul(nu, static_cast<RealType>(-2), s.data(), s.data());
233  sqrt(nu, s.data(), s.data());
234  mul(nu, const_pi_2<RealType>(), u2, u2);
235  sincos(nu, u2, u1, u2);
236  if (!is_one(stddev))
237  mul(nu, stddev, s.data(), s.data());
238  if (!is_zero(mean)) {
239  fma(nu, s.data(), u1, mean, u1);
240  fma(nu, s.data(), u2, mean, u2);
241  } else {
242  mul(nu, s.data(), u1, u1);
243  mul(nu, s.data(), u2, u2);
244  }
245 }

251 template <typename RealType, typename RNGType>
252 inline void normal_distribution(
253  RNGType &rng, std::size_t n, RealType *r, RealType mean, RealType stddev)
254 {
255  static_assert(std::is_floating_point<RealType>::value,
256  "**normal_distribution** USED WITH RealType OTHER THAN FLOATING POINT "
257  "TYPES");
258 
259  const std::size_t k = internal::BufferSize<RealType>::value;
260  const std::size_t m = n / k;
261  const std::size_t l = n % k;
262  for (std::size_t i = 0; i != m; ++i, r += k)
263  internal::normal_distribution_impl<k>(rng, k, r, mean, stddev);
264  internal::normal_distribution_impl<k>(rng, l, r, mean, stddev);
265  if (n % 2 != 0) {
266  U01Distribution<RealType> u01;
267  RealType u = u01(rng);
268  RealType v = u01(rng);
269  r[l - 1] = mean +
270  stddev * std::sqrt(-2 * std::log(u)) *
271  std::cos(const_pi_2<RealType>() * v);
272  }
273 }
