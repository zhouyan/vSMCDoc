343  static int cov_chol(std::size_t dim, const result_type *cov,
344  result_type *chol, MatrixLayout layout = RowMajor, bool upper = false,
345  bool packed = false)
346  {
347  unsigned l = layout == RowMajor ? 0 : 1;
348  unsigned u = upper ? 1 : 0;
349  unsigned p = packed ? 1 : 0;
350  unsigned c = (l << 2) + (u << 1) + p;
351  std::size_t k = 0;
352  switch (c) {
353  case 0: // Row, Lower, Full
354  k = 0;
355  for (std::size_t i = 0; i != dim; ++i)
356  for (std::size_t j = 0; j <= i; ++j)
357  chol[k++] = cov[i * dim + j];
358  break;
359  case 1: // Row, Lower, Pack
360  std::copy_n(cov, dim * (dim + 1) / 2, chol);
361  break;
362  case 2: // Row, Upper, Full
363  k = 0;
364  for (std::size_t i = 0; i != dim; ++i)
365  for (std::size_t j = 0; j <= i; ++j)
366  chol[k++] = cov[j * dim + i];
367  break;
368  case 3: // Row, Upper, Pack
369  k = 0;
370  for (std::size_t i = 0; i != dim; ++i)
371  for (std::size_t j = 0; j <= i; ++j)
372  chol[k++] = cov[dim * j - j * (j + 1) / 2 + i];
373  break;
374  case 4: // Col, Lower, Full
375  k = 0;
376  for (std::size_t i = 0; i != dim; ++i)
377  for (std::size_t j = 0; j <= i; ++j)
378  chol[k++] = cov[j * dim + i];
379  break;
380  case 5: // Col, Lower, Pack
381  k = 0;
382  for (std::size_t i = 0; i != dim; ++i)
383  for (std::size_t j = 0; j <= i; ++j)
384  chol[k++] = cov[dim * j - j * (j + 1) / 2 + i];
385  break;
386  case 6: // Col, Upper, Full
387  k = 0;
388  for (std::size_t j = 0; j != dim; ++j)
389  for (std::size_t i = 0; i <= j; ++i)
390  chol[k++] = cov[j * dim + i];
391  break;
392  case 7: // Col, Upper, Pack
393  std::copy_n(cov, dim * (dim + 1) / 2, chol);
394  break;
395  default: break;
396  }
397 
398  return static_cast<int>(LAPACKE_dpptrf(
399  LAPACK_ROW_MAJOR, 'L', static_cast<lapack_int>(dim), chol));
400  }

454 template <typename RealType, typename RNGType>
455 inline void normal_mv_distribution(RNGType &rng, std::size_t n, RealType *r,
456  std::size_t dim, const RealType *mean, const RealType *chol)
457 {
458  Vector<RealType> cholf(dim * dim);
459  for (std::size_t i = 0; i != dim; ++i)
460  for (std::size_t j = 0; j <= i; ++j)
461  cholf[i * dim + j] = *chol++;
462  normal_distribution(rng, n * dim, r, 0.0, 1.0);
463  internal::normal_mv_distribution_mulchol(n, r, dim, cholf.data());
464  for (std::size_t i = 0; i != n; ++i, r += dim)
465  add(dim, mean, r, r);
466 }
