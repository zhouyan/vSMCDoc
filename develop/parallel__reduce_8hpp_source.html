54 inline void parallel_reduce(const Range &range, WorkType &work)
55 {
56  std::vector<Range> range_vec(ThreadNum::instance().partition(range));
57  std::vector<WorkType> work_vec(range_vec.size(), work);
58  {
59  std::vector<ThreadGuard<std::thread>> tg;
60  tg.reserve(range_vec.size());
61  for (std::size_t i = 0; i != range_vec.size(); ++i) {
62  tg.push_back(ThreadGuard<std::thread>(
63  std::thread(std::ref(work_vec[i]), range_vec[i])));
64  }
65  }
66  for (std::size_t i = 0; i != work_vec.size(); ++i)
67  work.join(work_vec[i]);
68 }
