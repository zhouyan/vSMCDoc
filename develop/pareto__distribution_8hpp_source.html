187 template <typename RealType, typename RNGType>
188 inline void pareto_distribution(
189  RNGType &rng, std::size_t n, RealType *r, RealType a, RealType b)
190 {
191  const std::size_t k = 1000;
192  const std::size_t m = n / k;
193  const std::size_t l = n % k;
194  for (std::size_t i = 0; i != m; ++i)
195  internal::pareto_distribution_impl<k>(rng, k, r + i * k, a, b);
196  internal::pareto_distribution_impl<k>(rng, l, r + m * k, a, b);
197 }
