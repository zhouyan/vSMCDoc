218  void resize_by_range(size_type N, size_type first, size_type last)
219  {
220  VSMC_RUNTIME_ASSERT_CORE_PARTICLE_RESIZE_BY_RANGE;
221 
222  if (N == size_ && first == 0)
223  return;
224 
225  Vector<size_type> idx(static_cast<std::size_t>(N));
226  size_type M = last - first;
227  if (M >= N) {
228  for (size_type i = 0; i != N; ++i, ++first)
229  idx[static_cast<std::size_t>(i)] = first;
230  } else {
231  for (size_type i = 0; i != M; ++i, ++first)
232  idx[static_cast<std::size_t>(i)] = first;
233  resize_copy_index(static_cast<std::size_t>(N - M),
234  static_cast<std::size_t>(M), idx.data(), idx.data() + M);
235  }
236  resize(N, idx.data());

291  template <typename ResampleType>
292  bool resample(ResampleType &&op, double threshold)
293  {
294  std::size_t N = static_cast<std::size_t>(weight_.resample_size());
295  bool resampled = weight_.ess() < threshold * N;
296  if (resampled) {
297  const double *const rwptr = weight_.resample_data();
298  if (rwptr != nullptr) {
299  Vector<size_type> rep(static_cast<std::size_t>(N));
300  Vector<size_type> idx(static_cast<std::size_t>(N));
301  op(N, N, rng_, rwptr, rep.data());
302  resample_trans_rep_index(N, N, rep.data(), idx.data());
303  value_.copy(N, idx.data());
304  } else {
305  value_.copy(N, static_cast<const size_type *>(nullptr));
306  }
307  weight_.set_equal();
308  }
309 
310  return resampled;
311  }
