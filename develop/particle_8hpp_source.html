176  template <typename ResampleType>
177  bool resample(ResampleType &&op, double threshold)
178  {
179  std::size_t N = static_cast<std::size_t>(weight_.resample_size());
180  bool resampled = weight_.ess() < threshold * N;
181  if (resampled) {
182  const double *const rwptr = weight_.resample_data();
183  if (rwptr != nullptr) {
184 #if VSMC_USE_TBB
185  Vector<size_type> &rep = rep_.local();
186  Vector<size_type> &idx = idx_.local();
187  rep.resize(N);
188  idx.resize(N);
189 #else // VSMC_USE_TBB
190  Vector<size_type> rep(N);
191  Vector<size_type> idx(N);
192 #endif // VSMC_USE_TBB
193  op(N, N, rng_, rwptr, rep.data());
194  resample_trans_rep_index(N, N, rep.data(), idx.data());
195  value_.copy(N, idx.data());
196  } else {
197  value_.copy(N, static_cast<const size_type *>(nullptr));
198  }
199  weight_.set_equal();
200  }
201 
202  return resampled;
203  }
