245  void resize_by_range(size_type N, size_type first, size_type last)
246  {
247  VSMC_RUNTIME_ASSERT_CORE_PARTICLE_RESIZE_BY_RANGE;
248 
249  if (N == size_ && first == 0)
250  return;
251 
252  Vector<size_type> idx(static_cast<std::size_t>(N));
253  size_type M = last - first;
254  if (M >= N) {
255  for (size_type i = 0; i != N; ++i, ++first)
256  idx[static_cast<std::size_t>(i)] = first;
257  } else {

259  idx[static_cast<std::size_t>(i)] = first;

261  static_cast<std::size_t>(M), idx.data(), idx.data() + M);
