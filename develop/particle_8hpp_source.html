158  bool resample(const resample_type &op, double threshold)
159  {
160  std::size_t N = static_cast<std::size_t>(weight_.resample_size());
161  bool resampled = weight_.ess() < threshold * N;
162  if (resampled) {
163  const double *const rwptr = weight_.resample_data();
164  if (rwptr != nullptr) {
165 #if VSMC_USE_TBB
166  Vector<size_type> &rep = rep_.local();
167  Vector<size_type> &idx = idx_.local();
168  rep.resize(N);
169  idx.resize(N);
170 #else // VSMC_USE_TBB
171  Vector<size_type> rep(N);
172  Vector<size_type> idx(N);
173 #endif // VSMC_USE_TBB
174 
175  op(N, N, resample_rng_, rwptr, rep.data());
176  resample_trans_rep_index(N, N, rep.data(), idx.data());
177  value_.copy(N, idx.data());
178  } else {
179  value_.copy(N, static_cast<const size_type *>(nullptr));
180  }
181  weight_.set_equal();
182  }
183 
184  return resampled;
185  }
