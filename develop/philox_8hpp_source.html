150 template <std::size_t K, std::size_t I>
151 inline void philox_hilo(std::uint64_t b, std::uint64_t &hi, std::uint64_t &lo)
152 {
153  const std::uint64_t a = PhiloxRoundConstant<std::uint64_t, K, I>::value;
154  const unsigned whalf = 32;
155  const std::uint64_t lomask = (static_cast<std::uint64_t>(1) << whalf) - 1;
156 
157  lo = static_cast<std::uint64_t>(a * b);
158 
159  const std::uint64_t ahi = a >> whalf;
160  const std::uint64_t alo = a & lomask;
161  const std::uint64_t bhi = b >> whalf;
162  const std::uint64_t blo = b & lomask;
163 
164  const std::uint64_t ahbl = ahi * blo;
165  const std::uint64_t albh = alo * bhi;
166 
167  const std::uint64_t ahbl_albh = ((ahbl & lomask) + (albh & lomask));
168 
169  hi = ahi * bhi + (ahbl >> whalf) + (albh >> whalf);
170  hi += ahbl_albh >> whalf;
171  hi += ((lo >> whalf) < (ahbl_albh & lomask));
172 }
