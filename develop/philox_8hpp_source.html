158  void discard(result_type nskip)
159  {
160  std::size_t n = static_cast<std::size_t>(nskip);
161  if (index_ + n <= K) {
162  index_ += n;
163  return;
164  }
165 
166  n -= K - index_;
167  if (n <= K) {
168  index_ = K;
169  operator()();

175  index_ = K;
176  operator()();
