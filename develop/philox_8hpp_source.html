133 template <std::size_t K, std::size_t I>
134 inline void philox_hilo(std::uint64_t b, std::uint64_t &hi, std::uint64_t &lo)
135 {
136  const std::uint64_t a = PhiloxRoundConstant<std::uint64_t, K, I>::value;
137  const unsigned whalf = 32;
138  const std::uint64_t lomask = (static_cast<std::uint64_t>(1) << whalf) - 1;
139 
140  lo = static_cast<std::uint64_t>(a * b);
141 
142  const std::uint64_t ahi = a >> whalf;
143  const std::uint64_t alo = a & lomask;
144  const std::uint64_t bhi = b >> whalf;
145  const std::uint64_t blo = b & lomask;
146 
147  const std::uint64_t ahbl = ahi * blo;
148  const std::uint64_t albh = alo * bhi;
149 
150  const std::uint64_t ahbl_albh = ((ahbl & lomask) + (albh & lomask));
151 
152  hi = ahi * bhi + (ahbl >> whalf) + (albh >> whalf);
153  hi += ahbl_albh >> whalf;
154  hi += ((lo >> whalf) < (ahbl_albh & lomask));
155 }
