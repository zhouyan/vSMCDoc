163 template <std::size_t K, std::size_t I>
164 inline void philox_hilo(std::uint64_t b, std::uint64_t &hi, std::uint64_t &lo)
165 {
166  const std::uint64_t a = PhiloxRoundConstant<std::uint64_t, K, I>::value;
167  const unsigned whalf = 32;
168  const std::uint64_t lomask = (static_cast<std::uint64_t>(1) << whalf) - 1;
169 
170  lo = static_cast<std::uint64_t>(a * b);
171 
172  const std::uint64_t ahi = a >> whalf;
173  const std::uint64_t alo = a & lomask;
174  const std::uint64_t bhi = b >> whalf;
175  const std::uint64_t blo = b & lomask;
176 
177  const std::uint64_t ahbl = ahi * blo;
178  const std::uint64_t albh = alo * bhi;
179 
180  const std::uint64_t ahbl_albh = ((ahbl & lomask) + (albh & lomask));
181 
182  hi = ahi * bhi + (ahbl >> whalf) + (albh >> whalf);
183  hi += ahbl_albh >> whalf;
184  hi += ((lo >> whalf) < (ahbl_albh & lomask));
185 }
