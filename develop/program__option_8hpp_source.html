534  void process_arg_vector(Vector<std::string> &arg_vector, std::ostream &os)
535  {
536  Vector<std::pair<std::string, Vector<std::string>>> name_svals;
537  Vector<std::string> svals;
538  auto aiter = arg_vector.begin();
539  while (aiter != arg_vector.end() && !is_option(*aiter))
540  ++aiter;
541  while (aiter != arg_vector.end()) {
542  std::string name(aiter->begin() + 2, aiter->end());
543  ++aiter;
544  svals.clear();
545  while (aiter != arg_vector.end() && !is_option(*aiter)) {
546  svals.push_back(*aiter);
547  ++aiter;
548  }
549  name_svals.push_back(std::make_pair(name, svals));
550  }
551 
552  const std::string sval_true("1");
553  for (auto &nsv : name_svals) {
554  auto iter = option_find(nsv.first);
555  if (iter == option_vec_.end()) {
556  internal::program_option_warning(
557  nsv.first, "Unknown option", silent_, os);
558  continue;
559  }
560 
561  bool proc = false;
562  if (nsv.second.size() == 0 && std::get<1>(*iter)->is_bool()) {
563  proc = process_option(iter, sval_true, os);
564  } else if (nsv.second.size() == 0) {
565  internal::program_option_warning(
566  nsv.first, "No value found", silent_, os);
567  } else if (std::get<1>(*iter)->is_vector()) {
568  for (const auto &sval : nsv.second)
569  proc = process_option(iter, sval, os);
570  } else {
571  proc = process_option(iter, nsv.second.back(), os) || proc;
572  }
573  if (proc)
574  ++std::get<2>(*iter);
575  }
576 
577  for (auto &option : option_vec_)
578  if (std::get<2>(option) == 0)
579  if (std::get<1>(option)->set_default())
580  std::get<2>(option) = 1;
581 
582  if (help())
583  print_help(os);
584  }
