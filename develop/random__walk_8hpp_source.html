82  template <typename RNGType, typename LogTargetType, typename ProposalType>
83  std::size_t operator()(RNGType &rng, StateType &x, RealType *ltx,
84  LogTargetType &&log_target, ProposalType &&proposal)
85  {
86  StateType r;
87  RealType q = proposal(rng, x, r);
88  RealType s = ltx == nullptr ? log_target(x) : *ltx;
89  RealType t = log_target(r);
90  RealType p = t - s + q;
91  RealType u = std::log(runif_(rng));
92 
93  if (u < p) {
94  x = r;
95  if (ltx != nullptr)
96  *ltx = t;
97  return 1;
98  }
99  return 0;
100  }

104  RealType *ltx, LogTargetType &&log_target, ProposalType &&proposal)
105  {

252  RealType x, RealType &y, RealType z, RealType a, RealType b)

254  RealType r = std::exp(z) * (x - a) / (b - x);
