57 // Given N sorted U01 random variates
58 // Compute M replication numbers based on M weights
59 template <typename IntType, typename U01SeqType>
60 inline void inversion(std::size_t M, std::size_t N, const double *weight,
61  U01SeqType &u01seq, IntType *replication)
62 {
63  if (M == 0)
64  return;
65 
66  if (M == 1) {
67  *replication = static_cast<IntType>(N);
68  return;
69  }
70 
71  std::memset(replication, 0, sizeof(IntType) * M);
72 
73  if (N == 0)
74  return;
75 
76  std::size_t n = 0;
77  double accw = 0;
78  for (std::size_t i = 0; i != M - 1; ++i) {
79  accw += weight[i];
80  while (n != N && u01seq[n] <= accw) {
81  ++replication[i];
82  ++n;
83  }
84  }
85  replication[M - 1] = N - n;
86 }

88 // Given replication numbers, transfer them to copy_from index
89 template <typename IntType1, typename IntType2>
90 inline void cfrp_trans(std::size_t M, std::size_t N,
91  const IntType1 *replication, IntType2 *copy_from)
92 {
93  if (M == N) {
94  IntType1 time = 0;
95  IntType2 from = 0;
96  for (std::size_t to = 0; to != N; ++to) {
97  if (replication[to] != 0) {
98  copy_from[to] = static_cast<IntType2>(to);
99  } else {
100  // replication[to] has zero child, copy from elsewhere
101  if (replication[from] < time + 2) {
102  // only 1 child left on replication[from]
103  time = 0;
104  do // move from to a position with at least 2 children
105  ++from;
106  while (replication[from] < 2);
107  }
108  copy_from[to] = from;
109  ++time;
110  }
111  }
112  } else {
113  std::size_t to = 0;
114  for (std::size_t from = 0; from != M; ++from) {
115  const IntType1 rep = replication[from];
116  for (IntType1 r = 0; r != rep; ++r)
117  copy_from[to++] = static_cast<IntType2>(from);
118  }
119  }
120 }
