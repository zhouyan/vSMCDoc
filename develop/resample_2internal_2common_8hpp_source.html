57 // Given N sorted U01 random variates
58 // Compute M replication numbers based on M weights
59 template <typename IntType, typename U01SeqType>
60 inline void trans_usrp(std::size_t M, std::size_t N, const double *weight,
61  U01SeqType &&u01seq, IntType *replication)
62 {
63  if (M == 0)
64  return;
65 
66  if (M == 1) {
67  *replication = static_cast<IntType>(N);
68  return;
69  }
70 
71  std::memset(replication, 0, sizeof(IntType) * M);
72 
73  if (N == 0)
74  return;
75 
76  std::size_t n = 0;
77  double accw = 0;
78  for (std::size_t i = 0; i != M - 1; ++i) {
79  accw += weight[i];
80  while (n != N && u01seq[n] <= accw) {
81  ++replication[i];
82  ++n;
83  }
84  }
85  replication[M - 1] = N - n;
86 }

88 // Given replication numbers, transfer them to copy_from index
89 template <typename IntType1, typename IntType2>
90 inline void trans_rpcf(std::size_t M, std::size_t N,
91  const IntType1 *replication, IntType2 *copy_from)
92 {
93  if (M == 0 || N == 0)
94  return;
95 
96  if (M != N) {
97  std::size_t to = 0;
98  for (std::size_t from = 0; from != M; ++from) {
99  const IntType1 rep = replication[from];
100  for (IntType1 r = 0; r != rep; ++r)
101  copy_from[to++] = static_cast<IntType2>(from);
102  }
103  return;
104  }
105 
106  IntType1 time = 0;
107  IntType2 from = 0;
108  for (std::size_t to = 0; to != N; ++to) {
109  if (replication[to] != 0) {
110  copy_from[to] = static_cast<IntType2>(to);
111  } else {
112  // replication[to] has zero child, copy from elsewhere
113  if (replication[from] < time + 2) {
114  // only 1 child left on replication[from]
115  time = 0;
116  do // move from to a position with at least 2 children
117  ++from;
118  while (replication[from] < 2);
119  }
120  copy_from[to] = from;
121  ++time;
122  }
123  }
124 }

142  U01SeqType &&u01seq, IntType *copy_from)
143 {
144  if (M == 0 || N == 0)
145  return;

148 
149  if (M == 1)
150  return;

152  std::size_t n = 0;
153  double accw = 0;
154  for (std::size_t i = 0; i != M - 1; ++i) {
155  accw += weight[i];
156  while (n != N && u01seq[n] <= accw)
157  copy_from[n++] = static_cast<IntType>(i);
