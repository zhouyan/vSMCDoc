45  public:
46  template <typename IntType, typename RNGType>
47  void operator()(std::size_t M, std::size_t N, RNGType &rng,
48  const double *weight, IntType *replication)
49  {
50  residual_.resize(M);
51  integral_.resize(M);
52  for (std::size_t i = 0; i != M; ++i)
53  residual_[i] = std::modf(N * weight[i], integral_.data() + i);
54  double coeff = 1 / math::asum(M, residual_.data(), 1);
55  math::scal(M, coeff, residual_.data(), 1);
56 
57  IntType R = 0;
58  for (std::size_t i = 0; i != M; ++i)
59  R += static_cast<IntType>(integral_[i]);
60  std::size_t NN = N - static_cast<std::size_t>(R);
61  U01SequenceStratified<RNGType> u01seq(NN, rng);
62  resample_trans_u01_rep(M, NN, residual_.data(), u01seq, replication);
63  for (std::size_t i = 0; i != M; ++i)
64  replication[i] += static_cast<IntType>(integral_[i]);
65  }
66 
67  private:
68  Vector<double> residual_;
69  Vector<double> integral_;
70 }; // ResampleResidualStratified
