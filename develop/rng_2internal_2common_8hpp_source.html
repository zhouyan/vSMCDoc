286 {

295 template <typename T, std::size_t K>
296 inline void increment(
297  std::size_t n, std::array<T, K> &ctr, std::array<T, K> *ctr_block)
298 {
299  if (n == 0)
300  return;
301 
302  const std::uint64_t m =
303  static_cast<std::uint64_t>(std::numeric_limits<T>::max VSMC_MNE());
304  const std::uint64_t l = static_cast<std::uint64_t>(ctr.front());
305  const std::uint64_t k = static_cast<std::uint64_t>(n);
306  if (k < m && l < m - k) {
307  increment_safe(static_cast<T>(n), ctr, ctr_block);
308  } else {
309  for (std::size_t i = 0; i != n; ++i) {
310  increment(ctr);
311  ctr_block[i] = ctr;
312  }
313  }
314 }
