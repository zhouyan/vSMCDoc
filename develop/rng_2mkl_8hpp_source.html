157 template <MKL_INT BRNG, typename ResultType>
158 class MKLSkipAheadForce
159 {
160  public:
161  using size_type = MKL_INT;
162 
163  MKLSkipAheadForce() : buffer_size_(VSMC_RNG_MKL_VSL_BUFFER_SIZE) {}
164 
165  void operator()(MKLStream<BRNG> &stream, size_type nskip)
166  {
167  if (nskip == 0)
168  return;
169 
170  if (nskip < buffer_size_) {
171  if (buffer_.size() < nskip)
172  buffer_.resize(nskip);
173  uniform_bits_(stream, nskip, buffer_.data());
174  } else {
175  buffer_.resize(buffer_size_);
176  size_type repeat = nskip / buffer_size_;
177  size_type remain = nskip - repeat * buffer_size_;
178  for (size_type r = 1; r != repeat + 1; ++r) {
179  size_type n = r * buffer_size_;
180  uniform_bits_(stream, n, buffer_.data());
181  }
182  uniform_bits_(stream, remain, buffer_.data());
183  }
184  }
185 
186  void buffer_size(MKL_INT size)
187  {
188  buffer_size_ = size > 0 ? size : VSMC_RNG_MKL_VSL_BUFFER_SIZE;
189  }
190 
191  MKL_INT buffer_size() { return buffer_size_; }
192 
193  private:
194  Vector<ResultType> buffer_;
195  typename MKLUniformBitsTrait<BRNG, ResultType>::type uniform_bits_;
196  MKL_INT buffer_size_;
197 }; // class SkipAheadForce

395 template <typename ResultType, typename Derived>
396 class MKLDistribution
397 {
398  public:
399  using result_type = ResultType;
400 
401  MKLDistribution()
402  : buffer_size_(VSMC_RNG_MKL_VSL_BUFFER_SIZE), index_(buffer_size_)
403  {
404  }
405 
406  template <MKL_INT BRNG>
407  result_type operator()(MKLStream<BRNG> &stream)
408  {
409  if (index_ == buffer_size_) {
410  buffer_.resize(static_cast<std::size_t>(buffer_size_));
411  static_cast<Derived *>(this)->generate(
412  stream, buffer_size_, buffer_.data());
413  index_ = 0;
414  }
415 
416  return buffer_[index_++];
417  }
418 
419  template <MKL_INT BRNG>
420  void operator()(MKLStream<BRNG> &stream, MKL_INT n, result_type *r)
421  {
422  static_cast<Derived *>(this)->generate(stream, n, r);
423  }
424 
425  void reset() { index_ = buffer_size_; }
426 
428  void buffer_size(MKL_INT size)
429  {
430  buffer_size_ = size > 0 ? size : VSMC_RNG_MKL_VSL_BUFFER_SIZE;
431  }
432 
433  MKL_INT buffer_size() const { return buffer_size_; }
434 
435  protected:
436  template <MKL_INT BRNG>
437  void generate_error_check(int status, const char *name)
438  {
439  if (status == VSL_ERROR_OK)
440  return;
441 
442  std::string dist_name(name);
443  dist_name = "MKL" + dist_name + "Distribution::generate";
444  std::string mkl_vsl_name(
445  mkl_vsl_name_prefix(static_cast<result_type>(0)));
446  mkl_vsl_name += "Rng";
447  mkl_vsl_name += name;
448  internal::mkl_error_check(
449  status, dist_name.c_str(), mkl_vsl_name.c_str());
450  }
451 
452  private:
453  Vector<result_type> buffer_;
454  MKL_INT buffer_size_;
455  MKL_INT index_;
456 
457  std::string mkl_vsl_name_prefix(MKL_INT) { return "vi"; }
458  std::string mkl_vsl_name_prefix(unsigned) { return "vi"; }
459  std::string mkl_vsl_name_prefix(unsigned MKL_INT64) { return "vi"; }
460  std::string mkl_vsl_name_prefix(float) { return "vs"; }
461  std::string mkl_vsl_name_prefix(double) { return "vd"; }
462 }; // class MKLDistribution
