156 template <MKL_INT BRNG, typename ResultType>
157 struct MKLSkipAheadForce {
158  typedef MKL_INT size_type;
159 
160  MKLSkipAheadForce() : buffer_size_(VSMC_RNG_MKL_VSL_BUFFER_SIZE) {}
161 
162  void operator()(MKLStream<BRNG> &stream, size_type nskip)
163  {
164  if (nskip == 0)
165  return;
166 
167  if (nskip < buffer_size_) {
168  if (buffer_.size() < nskip)
169  buffer_.resize(nskip);
170  uniform_bits_(stream, nskip, buffer_.data());
171  } else {
172  buffer_.resize(buffer_size_);
173  size_type repeat = nskip / buffer_size_;
174  size_type remain = nskip - repeat * buffer_size_;
175  for (size_type r = 1; r != repeat + 1; ++r) {
176  size_type n = r * buffer_size_;
177  uniform_bits_(stream, n, buffer_.data());
178  }
179  uniform_bits_(stream, remain, buffer_.data());
180  }
181  }
182 
183  void buffer_size(MKL_INT size)
184  {
185  buffer_size_ = size > 0 ? size : VSMC_RNG_MKL_VSL_BUFFER_SIZE;
186  }
187 
188  MKL_INT buffer_size() { return buffer_size_; }
189 
190  private:
191  AlignedVector<ResultType> buffer_;
192  typename traits::MKLUniformBitsTrait<BRNG, ResultType>::type uniform_bits_;
193  MKL_INT buffer_size_;
194 }; // strut SkipAheadForce

381 template <typename ResultType, typename Derived>
382 class MKLDistribution
383 {
384  public:
385  typedef ResultType result_type;
386 
387  MKLDistribution()
388  : buffer_size_(VSMC_RNG_MKL_VSL_BUFFER_SIZE), index_(buffer_size_)
389  {
390  }
391 
392  template <MKL_INT BRNG>
393  result_type operator()(MKLStream<BRNG> &stream)
394  {
395  if (index_ == buffer_size_) {
396  buffer_.resize(static_cast<std::size_t>(buffer_size_));
397  static_cast<Derived *>(this)->generate(
398  stream, buffer_size_, buffer_.data());
399  index_ = 0;
400  }
401 
402  return buffer_[index_++];
403  }
404 
405  template <MKL_INT BRNG>
406  void operator()(MKLStream<BRNG> &stream, MKL_INT n, result_type *r)
407  {
408  static_cast<Derived *>(this)->generate(stream, n, r);
409  }
410 
411  void reset() { index_ = buffer_size_; }
412 
414  void buffer_size(MKL_INT size)
415  {
416  buffer_size_ = size > 0 ? size : VSMC_RNG_MKL_VSL_BUFFER_SIZE;
417  }
418 
419  MKL_INT buffer_size() const { return buffer_size_; }
420 
421  protected:
422  template <MKL_INT BRNG>
423  void generate_error_check(int status, const char *name)
424  {
425  if (status == VSL_ERROR_OK)
426  return;
427 
428  std::string dist_name(name);
429  dist_name = "MKL" + dist_name + "Distribution::generate";
430  std::string mkl_vsl_name(
431  mkl_vsl_name_prefix(static_cast<result_type>(0)));
432  mkl_vsl_name += "Rng";
433  mkl_vsl_name += name;
434  internal::mkl_error_check(
435  status, dist_name.c_str(), mkl_vsl_name.c_str());
436  }
437 
438  private:
439  AlignedVector<result_type> buffer_;
440  MKL_INT buffer_size_;
441  MKL_INT index_;
442 
443  std::string mkl_vsl_name_prefix(MKL_INT) { return "vi"; }
444  std::string mkl_vsl_name_prefix(unsigned) { return "vi"; }
445  std::string mkl_vsl_name_prefix(unsigned MKL_INT64) { return "vi"; }
446  std::string mkl_vsl_name_prefix(float) { return "vs"; }
447  std::string mkl_vsl_name_prefix(double) { return "vd"; }
448 }; // class MKLDistribution
