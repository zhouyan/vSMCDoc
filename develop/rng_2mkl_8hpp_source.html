238  void seed(MKL_UINT s, MKL_INT offset)
239  {
240  typename internal::MKLOffset<BRNG>::type off;
241  off.set(offset);
242  stream_.reset(BRNG + off.get(), s);
243  index_ = M_;
244  }
245 
246  result_type operator()()
247  {
248  if (index_ == M_) {
249  internal::MKLUniformBits<Bits>::eval(
250  stream_, static_cast<MKL_INT>(M_), buffer_.data());
251  index_ = 0;
252  }
253 
254  return buffer_[index_++];
255  }

257  void discard(long long nskip)
258  {
259  internal::MKLDiscard<BRNG, Bits>::eval(stream_, nskip);
260  index_ = M_;
261  }
