770  void do_acch()
771  {
772  if (accept_history_.empty())
773  accept_history_.push_back(Vector<std::size_t>());
774  std::size_t acc_size = move_queue_.size() + mcmc_queue_.size();
775  if (accept_size() < acc_size) {
776  std::size_t diff = acc_size - accept_size();
777  for (std::size_t d = 0; d != diff; ++d)
778  accept_history_.push_back(Vector<std::size_t>());
779  }
780  for (std::size_t i = 0; i != accept_size(); ++i)
781  accept_history_[i].resize(iter_size());
782  }

877  Vector<std::pair<std::size_t, const Monitor<T> *>> miter;
878  for (const auto &m : monitor_)

880  miter.push_back(std::make_pair(0, &m.second));
881 
882  for (std::size_t iter = 0; iter != iter_size(); ++iter) {
883  *first++ = ess_history_[iter];

887  iter == m.second->index(m.first)) {
888  for (std::size_t id = 0; id != md; ++id, ++first)
889  *first = m.second->record(id, m.first);
890  ++m.first;
891  } else {
892  first = std::fill_n(first, md, missing_data);
893  }
894  }
895  }
