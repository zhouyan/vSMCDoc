232  template <typename RNGType>
233  result_type standard_1(RNGType &rng) const
234  {
235  U01DistributionType<RNGType, RealType> runif;
236  result_type w = -std::log(1 - runif(rng));
237  result_type u = (runif(rng) - 0.5) * math::pi<result_type>();
238  result_type a =
239  (math::pi_by2<result_type>() + param_.skewness_ * u) * std::tan(u);
240  result_type b =
241  std::log(math::pi_by2<result_type>() * w * std::cos(u));
242  result_type c =
243  std::log(math::pi_by2<result_type>() + param_.skewness_ * u);
244  result_type x = (a - param_.skewness_ * (b - c)) / param_.xi_;
245 
246  return x;
247  }
248 
249  template <typename RNGType>
250  result_type standard_a(RNGType &rng) const
251  {
252  U01DistributionType<RNGType, RealType> runif;
253  result_type w = -std::log(1 - runif(rng));
254  result_type u = (runif(rng) - 0.5) * math::pi<result_type>();
255  result_type a = 0.5 * std::log(1 + param_.zeta_ * param_.zeta_) /
256  param_.stability_;
257  result_type b = std::sin(param_.stability_ * (u + param_.xi_));
258  result_type c = std::log(std::cos(u)) / param_.stability_;
259  result_type d = (1 - param_.stability_) / param_.stability_ *
260  std::log(std::cos(u - param_.stability_ * (u + param_.xi_)) / w);
261  result_type x = b * std::exp(a - c + d);
262 
263  return x;
264  }

267  {

271  }
