180  template <MatrixOrder ROrder, typename OutputIter>
181  void read_state_matrix(OutputIter first) const
182  {
183  if (ROrder == Order) {
184  std::copy(data_.begin(), data_.end(), first);
185  } else {
186  const StateMatrix<Order, Dim, T> *sptr =
187  static_cast<const StateMatrix<Order, Dim, T> *>(this);
188  if (ROrder == RowMajor) {
189  for (size_type i = 0; i != size_; ++i) {
190  for (std::size_t d = 0; d != this->dim(); ++d) {
191  *first = sptr->state(i, d);
192  ++first;
193  }
194  }
