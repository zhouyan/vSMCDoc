166  void read_state_matrix(OutputIter first) const
167  {
168  if (ROrder == Order) {

170  } else {
171  const StateMatrix<Order, Dim, T> *sptr =
172  static_cast<const StateMatrix<Order, Dim, T> *>(this);
173  if (ROrder == RowMajor) {
174  for (size_type i = 0; i != size_; ++i) {
175  for (std::size_t d = 0; d != this->dim(); ++d) {
176  *first = sptr->state(i, d);

178  }
179  }
