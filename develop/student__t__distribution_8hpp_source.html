185 template <typename RealType, typename RNGType>
186 inline void student_t_distribution(
187  RNGType &rng, std::size_t n, RealType *r, RealType df)
188 {
189  const std::size_t k = 1000;
190  const std::size_t m = n / k;
191  const std::size_t l = n % k;
192  for (std::size_t i = 0; i != m; ++i)
193  internal::student_t_distribution_impl<k>(rng, k, r + i * k, df);
194  internal::student_t_distribution_impl<k>(rng, l, r + m * k, df);
195 }
