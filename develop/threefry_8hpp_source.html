163  void discard(result_type nskip)
164  {
165  std::size_t n = static_cast<std::size_t>(nskip);
166  if (index_ + n <= K) {
167  index_ += n;
168  return;
169  }
170 
171  n -= K - index_;
172  if (n <= K) {
173  index_ = K;
174  operator()();
175  index_ = n;
176  return;
177  }
178 
179  internal::increment(ctr_, static_cast<result_type>(n / K));
180  index_ = K;
181  operator()();
182  index_ = n % K;
183  }
