328 template <typename ResultType, std::size_t K,

330 class ThreefryEngine
331 {
332  public:
333  using result_type = ResultType;
334  using key_type = std::array<ResultType, K>;
335  using ctr_type = std::array<ResultType, K>;

337  explicit ThreefryEngine(result_type s = 0) : index_(K)
338  {

343  template <typename SeedSeq>
344  explicit ThreefryEngine(SeedSeq &seq,
345  typename std::enable_if<internal::is_seed_seq<SeedSeq, result_type,
346  key_type, ThreefryEngine<ResultType, K, Rounds>>::value>::type * =
347  nullptr)

352  }
353 
354  ThreefryEngine(const key_type &k) : index_(K)
355  {
356  VSMC_STATIC_ASSERT_RNG_THREEFRY();

358  }
359 
360  void seed(result_type s)
361  {
362  ctr_.fill(0);
363  key_type k;
364  k.fill(0);
365  k.front() = s;
366  internal::ThreefryInitPar<ResultType, K>::eval(k, par_);
367  index_ = K;
368  }
369 
370  template <typename SeedSeq>
371  void seed(SeedSeq &seq,
372  typename std::enable_if<internal::is_seed_seq<SeedSeq, result_type,
373  key_type, ThreefryEngine<ResultType, K, Rounds>>::value>::type * =
374  nullptr)
375  {
376  ctr_.fill(0);
377  key_type k;
378  seq.generate(k.begin(), k.end());
379  internal::ThreefryInitPar<ResultType, K>::eval(k, par_);
380  index_ = K;
381  }
382 
383  void seed(const key_type &k)
384  {
385  ctr_.fill(0);
386  internal::ThreefryInitPar<ResultType, K>::eval(k, par_);
387  index_ = K;
388  }
389 
390  ctr_type ctr() const { return ctr_; }
391 
392  key_type key() const
393  {
394  key_type k;
395  for (std::size_t i = 0; i != K; ++i)
396  k[i] = par_[i];
397 
398  return k;
399  }
400 
401  void ctr(const ctr_type &c)
402  {
403  ctr_ = c;
404  index_ = K;
405  }
406 
407  void key(const key_type &k)
408  {
409  internal::ThreefryInitPar<ResultType, K>::eval(k, par_);
410  index_ = K;
411  }
412 
413  result_type operator()()
414  {
415  if (index_ == K) {
416  generate_buffer();
417  index_ = 0;
418  }
419 
420  return buffer_[index_++];
421  }
422 
423  void discard(result_type nskip)
424  {
425  std::size_t n = static_cast<std::size_t>(nskip);
426  if (index_ + n <= K) {
427  index_ += n;
428  return;
429  }
430 
431  n -= K - index_;
432  if (n <= K) {
433  index_ = K;
434  operator()();
435  index_ = n;
436  return;
437  }
438 
439  internal::increment(ctr_, static_cast<result_type>(n / K));
440  index_ = K;
441  operator()();
442  index_ = n % K;
443  }
