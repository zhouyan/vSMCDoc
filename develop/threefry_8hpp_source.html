399  void discard(result_type nskip)
400  {
401  std::size_t n = static_cast<std::size_t>(nskip);
402  if (index_ + n <= K) {
403  index_ += n;
404  return;
405  }
406 
407  n -= K - index_;
408  if (n <= K) {
409  index_ = K;
410  operator()();
411  index_ = n;
412  return;
413  }
414 
415  counter::increment(ctr_, static_cast<result_type>(n / K));
416  index_ = K;
417  operator()();
418  index_ = n % K;
419  }
