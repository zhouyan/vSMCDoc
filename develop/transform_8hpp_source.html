48 template <typename IntType, typename U01SeqType>
49 inline void resample_trans_u01_rep(std::size_t N, std::size_t M,
50  const double *weight, U01SeqType &&u01seq, IntType *replication)
51 {
52  if (N == 0)
53  return;
54 
55  if (N == 1) {
56  *replication = static_cast<IntType>(M);
57  return;
58  }
59 
60  std::memset(replication, 0, sizeof(IntType) * N);
61  if (M == 0)
62  return;
63 
64  double accw = 0;
65  std::size_t j = 0;
66  for (std::size_t i = 0; i != N - 1; ++i) {
67  accw += weight[i];
68  while (j != M && u01seq[j] < accw) {
69  ++replication[i];
70  ++j;
71  }
72  }
73  replication[N - 1] = static_cast<IntType>(M - j);
74 }

83 template <typename IntType1, typename IntType2>
84 inline void resample_trans_rep_index(
85  std::size_t N, std::size_t M, const IntType1 *replication, IntType2 *index)
86 {
87  if (N == 0 || M == 0)
88  return;
89 
90  const std::size_t K = std::min(N, M);
91  IntType1 time = 0;
92  std::size_t src = 0;
93 
94  auto good_src = [K, replication, &time, &src]() {
95  return src < K ? replication[src] > time + 1 : replication[src] > time;
96  };
97 
98  auto find_src = [&time, &src, &good_src]() {
99  if (!good_src()) {
100  time = 0;
101  do
102  ++src;
103  while (!good_src());
104  }
105  };
106 
107  for (std::size_t dst = 0; dst != K; ++dst) {
108  if (replication[dst] > 0) {
109  index[dst] = static_cast<IntType2>(dst);
110  } else {
111  find_src();
112  index[dst] = static_cast<IntType2>(src);
113  ++time;
114  }
115  }
116 
117  for (std::size_t dst = K; dst < M; ++dst) {
118  find_src();
119  index[dst] = static_cast<IntType2>(src);
120  ++time;
121  }
122 }

133 template <typename IntType>
134 inline std::size_t resample_trans_residual(std::size_t N, std::size_t M,
135  const double *weight, double *resid, IntType *integ)
136 {
137  double integral = 0;
138  const double coeff = static_cast<double>(M);
139  for (std::size_t i = 0; i != N; ++i) {
140  resid[i] = std::modf(coeff * weight[i], &integral);
141  integ[i] = static_cast<IntType>(integral);
142  }
143  mul(N, 1 / std::accumulate(resid, resid + N, 0.0), resid, resid);
144  IntType R = std::accumulate(integ, integ + N, static_cast<IntType>(0));
145 
146  return M - static_cast<std::size_t>(R);
147 }
