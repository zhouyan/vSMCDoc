62 template <typename RealType, typename RNGType>
63 inline void u01_distribution_impl_u64(
64  RNGType &rng, std::size_t n, RealType *r, typename RNGType::result_type *s)
65 {
66  rng_rand(rng, n / 2, s);
67  std::uint32_t *u = reinterpret_cast<std::uint32_t *>(s);
68  if (n % 2 != 0)
69  u[n - 1] = static_cast<std::uint32_t>(rng());
70  for (std::size_t i = 0; i != n; ++i)
71  r[i] = U01<std::uint32_t, RealType, Closed, Open>::eval(u[i]);
72 }

74 template <typename RealType, typename RNGType>
75 inline void u01_distribution_impl(
76  RNGType &rng, std::size_t n, RealType *r, std::true_type, std::false_type)
77 {
78  const std::size_t k = 1000;
79  const std::size_t m = n / k;
80  const std::size_t l = n % k;
81  typename RNGType::result_type s[k];
82  for (std::size_t i = 0; i != m; ++i)
83  u01_distribution_impl_u32(rng, k, r + i * k, s);
84  u01_distribution_impl_u32(rng, l, r + m * k, s);
85 }
