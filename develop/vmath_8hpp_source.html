130 template <typename T>
131 inline void linear_frac(std::size_t n, const T *a, const T *b, T beta_a,
132  T beta_b, T mu_a, T mu_b, T *y)
133 {
134  const std::size_t k = 1000;
135  const std::size_t m = n / k;
136  const std::size_t l = n % k;
137  for (std::size_t i = 0; i != m; ++i, a += k, y += k) {
138  for (std::size_t j = 0; j != k; ++j)
139  y[j] = beta_a * a[j] + mu_a;
140  for (std::size_t j = 0; j != k; ++j)
141  y[j] /= beta_b * b[j] + mu_b;
142  }
143  for (std::size_t i = 0; i != l; ++i)
144  y[i] = beta_a * a[i] + mu_a;
145  for (std::size_t i = 0; i != l; ++i)
146  y[i] /= beta_b * b[i] + mu_b;
147 }
