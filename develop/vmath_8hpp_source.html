244 template <typename T>
245 inline void linear_frac(std::size_t n, const T *a, const T *b, T beta_a,
246  T beta_b, T mu_a, T mu_b, T *y)
247 {
248  const std::size_t k = internal::BufferSize<T>::value;
249  const std::size_t m = n / k;
250  const std::size_t l = n % k;
251  for (std::size_t i = 0; i != m; ++i, a += k, y += k) {
252  for (std::size_t j = 0; j != k; ++j)
253  y[j] = beta_a * a[j] + mu_a;
254  for (std::size_t j = 0; j != k; ++j)
255  y[j] /= beta_b * b[j] + mu_b;
256  }
257  for (std::size_t i = 0; i != l; ++i)
258  y[i] = beta_a * a[i] + mu_a;
259  for (std::size_t i = 0; i != l; ++i)
260  y[i] /= beta_b * b[i] + mu_b;
261 }
