315 inline void vLinearFrac (std::size_t n, const T *a, const T *b,
316  T beta_a, T beta_b, T mu_a, T mu_b, T *y)
317 {
318  for (std::size_t i = 0; i != n; ++i)
319  y[i] = beta_a * a[i] + mu_a;
320  for (std::size_t i = 0; i != n; ++i)
321  y[i] /= beta_b * b[i] + mu_b;
322 }
