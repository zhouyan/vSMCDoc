184 template <typename RealType, typename RNGType>
185 inline void weibull_distribution_impl(
186  RNGType &rng, std::size_t n, RealType *r, RealType a, RealType b)
187 {
188  u01_oo_distribution(rng, n, r);
189  log(n, r, r);
190  if (is_equal<RealType>(a, 1)) {
191  mul(n, -b, r, r);
192  } else {
193  mul(n, static_cast<RealType>(-1), r, r);
194  log(n, r, r);
195  mul(n, 1 / a, r, r);
196  exp(n, r, r);
197  mul(n, b, r, r);
198  }
199 }

205 template <typename RealType, typename RNGType>
206 inline void weibull_distribution(
207  RNGType &rng, std::size_t n, RealType *r, RealType a, RealType b)
208 {
209  const std::size_t k = 1000;
210  const std::size_t m = n / k;
211  const std::size_t l = n % k;
212  for (std::size_t i = 0; i != m; ++i)
213  internal::weibull_distribution_impl(rng, k, r + i * k, a, b);
214  internal::weibull_distribution_impl(rng, l, r + m * k, a, b);
215 }
