316  virtual double compute_ess(const double *first, bool use_log) const
317  {
318  Vector<double> buffer(size_);
319  if (use_log) {
320  math::vLn(size_, data_.data(), buffer.data());
321  math::vAdd(size_, buffer.data(), first, buffer.data());
322  double wmax = *(std::max_element(buffer.begin(), buffer.end()));
323  for (size_type i = 0; i != size_; ++i)
324  buffer[i] -= wmax;
325  math::vExp(size_, buffer.data(), buffer.data());
326  } else {
327  math::vMul(size_, data_.data(), first, buffer.data());
328  }
329 
330  math::scal(
331  size_, 1 / math::asum(size_, buffer.data(), 1), buffer.data(), 1);
332

334  }

337  virtual double compute_cess(const double *first, bool use_log) const
338  {
339  Vector<double> buffer;
340  if (use_log) {
341  buffer.resize(size_);
342  math::vExp(size_, first, buffer.data());
343  }
344  const double *const bptr = use_log ? buffer.data() : first;
345 
346  double above = 0;
347  double below = 0;
348  for (size_type i = 0; i != size_; ++i) {
349  double wb = data_[i] * bptr[i];
350  above += wb;
351  below += wb * bptr[i];
352  }
353 
354  return above * above / below;
355  }
