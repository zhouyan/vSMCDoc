234  double normalize(bool use_log)
235  {
236  double *w = data_.data();
237  double accw = 0;

239  const std::size_t m = size() / k;

241  for (std::size_t i = 0; i != m; ++i, w += k)
242  normalize_eval(k, w, accw, use_log);
