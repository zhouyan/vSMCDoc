264  double normalize(bool use_log)
265  {
266  double *w = data_.data();
267  double accw = 0;
268  const std::size_t k = 1024;
269  const std::size_t m = size() / k;
270  const std::size_t l = size() % k;
271  for (std::size_t i = 0; i != m; ++i, w += k)
272  normalize_eval(k, w, accw, use_log);
273  normalize_eval(l, w, accw, use_log);
274  ::vsmc::mul(size(), 1 / accw, data_.data(), data_.data());
275 
276  return 1 / cblas_ddot(static_cast<VSMC_CBLAS_INT>(size()),
277  data_.data(), 1, data_.data(), 1);
