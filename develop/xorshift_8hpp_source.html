170 template <unsigned A, unsigned B, unsigned C, unsigned, typename ResultType,
171  std::size_t R, std::size_t S>
172 inline ResultType xorshift(
173  std::array<ResultType, 1> &state, XorshiftIndex<ResultType, 1, R, S> &)
174 {
175  state.front() ^= (state.front()) << A;
176  state.front() ^= (state.front()) >> B;
177  state.front() ^= (state.front()) << C;
178 
179  return state.front();
180 }

182 template <unsigned A, unsigned B, unsigned C, unsigned D, typename ResultType,
183  std::size_t K, std::size_t R, std::size_t S>
184 inline ResultType xorshift(std::array<ResultType, K> &state,
185  XorshiftIndex<ResultType, K, R, S> &index)
186 {
187  ResultType xr = state[index.r()];
188  xr = XorshiftLeft<A != 0, ResultType, A>::shift(xr);
189  xr = XorshiftRight<B != 0, ResultType, B>::shift(xr);
190 
191  ResultType xs = state[index.s()];
192  xs = XorshiftLeft<C != 0, ResultType, C>::shift(xs);
193  xs = XorshiftRight<D != 0, ResultType, D>::shift(xs);
194 
195  index.shift(state);
196 
197  return state[index.k()] = xs ^ xr;
198 }
