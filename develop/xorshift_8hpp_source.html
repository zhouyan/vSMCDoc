154 template <unsigned A, unsigned B, unsigned C, unsigned D,
155  typename ResultType, std::size_t K, std::size_t R, std::size_t S>
156 inline ResultType xorshift (Array<ResultType, K> &state,
157  XorshiftIndex<ResultType, K, R, S> &index)
158 {
159  ResultType xr = state[index.r()];
160  xr = XorshiftLeft <A != 0, ResultType, A>::shift(xr);
161  xr = XorshiftRight<B != 0, ResultType, B>::shift(xr);
162 
163  ResultType xs = state[index.s()];
164  xs = XorshiftLeft <C != 0, ResultType, C>::shift(xs);
165  xs = XorshiftRight<D != 0, ResultType, D>::shift(xs);
166 
167  index.shift(state);
168 
169  return state[index.k()] = xs^xr;
170 }
