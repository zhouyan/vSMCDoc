222 template <unsigned A, unsigned B, unsigned C, unsigned, typename ResultType,
223  std::size_t R, std::size_t S>
224 inline ResultType xorshift(
225  std::array<ResultType, 1> &state, XorshiftIndex<ResultType, 1, R, S> &)
226 {
227  state.front() ^= (state.front()) << A;
228  state.front() ^= (state.front()) >> B;
229  state.front() ^= (state.front()) << C;
230 
231  return state.front();
232 }

234 template <unsigned A, unsigned B, unsigned C, unsigned D, typename ResultType,
235  std::size_t K, std::size_t R, std::size_t S>
236 inline ResultType xorshift(std::array<ResultType, K> &state,
237  XorshiftIndex<ResultType, K, R, S> &index)
238 {
239  ResultType xr = state[index.r()];
240  xr = XorshiftLeft<A != 0, ResultType, A>::eval(xr);
241  xr = XorshiftRight<B != 0, ResultType, B>::eval(xr);
242 
243  ResultType xs = state[index.s()];
244  xs = XorshiftLeft<C != 0, ResultType, C>::eval(xs);
245  xs = XorshiftRight<D != 0, ResultType, D>::eval(xs);
246 
247  index.eval(state);
248 
249  return state[index.k()] = xs ^ xr;
250 }
