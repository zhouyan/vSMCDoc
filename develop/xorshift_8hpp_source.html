240 template <unsigned A, unsigned B, unsigned C, unsigned, typename ResultType,
241  std::size_t R, std::size_t S>
242 inline ResultType xorshift(
243  std::array<ResultType, 1> &state, XorshiftIndex<ResultType, 1, R, S> &)
244 {
245  state.front() ^= (state.front()) << A;
246  state.front() ^= (state.front()) >> B;
247  state.front() ^= (state.front()) << C;
248 
249  return state.front();
250 }

252 template <unsigned A, unsigned B, unsigned C, unsigned D, typename ResultType,
253  std::size_t K, std::size_t R, std::size_t S>
254 inline ResultType xorshift(std::array<ResultType, K> &state,
255  XorshiftIndex<ResultType, K, R, S> &index)
256 {
257  ResultType xr = state[index.r()];
258  xr = XorshiftLeft<A != 0, ResultType, A>::shift(xr);
259  xr = XorshiftRight<B != 0, ResultType, B>::shift(xr);
260 
261  ResultType xs = state[index.s()];
262  xs = XorshiftLeft<C != 0, ResultType, C>::shift(xs);
263  xs = XorshiftRight<D != 0, ResultType, D>::shift(xs);
264 
265  index.shift(state);
266 
267  return state[index.k()] = xs ^ xr;
268 }
