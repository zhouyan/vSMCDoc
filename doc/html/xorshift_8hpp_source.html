145  typename ResultType, std::size_t K, std::size_t R, std::size_t S>
146 inline ResultType xorshift (Array<ResultType, K> &state,
147  XorshiftIndex<ResultType, K, R, S> &index)
148 {
149  ResultType xr = state[index.r()];
150  xr = XorshiftLeft <A != 0, ResultType, A>::shift(xr);
