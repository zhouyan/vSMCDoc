513  _mm_castsi128_ps(xmm6_),
514  _mm_castsi128_ps(xmm1_), 0xF0));
515  xmm6_ = _mm_xor_si128 (xmm6_, xmm5_); // pxor xmm6, xmm5
516  xmm4_ = _mm_xor_si128 (xmm4_, xmm6_); // pxor xmm4, xmm6
517  xmm7_ = _mm_shuffle_epi32(xmm4_, 0x0E); // pshufd xmm7, xmm4, 0x0E
