467  void setup_cl_manager (cl_device_type dev_type)
468  {
469  setup_ = false;
470 
471  bool setup_platform = platform_filter(dev_type);
472  VSMC_RUNTIME_WARNING_CL_MANAGER_SETUP_PLATFORM;
473  if (!setup_platform) return;
474 
475  bool setup_context = false;
476  bool setup_device = false;
477  try {
478  std::vector<cl::Device> dev_pool;
479  std::vector<cl::Device> dev_select;
480  platform_.getDevices(dev_type, &dev_pool);
481  device_filter(dev_pool, dev_select);
482  if (dev_select.size() != 0) {
483  cl_context_properties context_properties[] = {
484  static_cast<cl_context_properties>(CL_CONTEXT_PLATFORM),
485  reinterpret_cast<cl_context_properties>(platform_()), 0
486  };
487  context_ = cl::Context(dev_select, context_properties);
488  setup_context = true;
489  device_vec_ = context_.getInfo<CL_CONTEXT_DEVICES>();
490  device_ = device_vec_[0];
491  setup_device = true;
492  }
493  } catch (cl::Error) {}
494  VSMC_RUNTIME_WARNING_CL_MANAGER_SETUP_CONTEXT;
495  VSMC_RUNTIME_WARNING_CL_MANAGER_SETUP_DEVICE;
496  if (!setup_context) return;
497  if (!setup_device) return;
498 
499  bool setup_command_queue = false;
500  try {
501  command_queue_ = cl::CommandQueue(context_, device_, 0);
502  setup_command_queue = true;
503  } catch (cl::Error) {}
504  VSMC_RUNTIME_WARNING_CL_MANAGER_SETUP_COMMAND_QUEUE;
505  if (!setup_command_queue) return;
506 
507  setup_ = true;
508  }

510  bool platform_filter (cl_device_type dev_type)
511  {
512  std::vector<cl::Platform> platform_vec;
513  try {
514  cl::Platform::get(&platform_vec);
515  } catch (cl::Error) {
516  platform_vec.clear();
517  }
518  if (platform_vec.size() == 0)
519  return false;
520 
521  // If not using default platform
522  if (!setup_default_.default_platform()) {
523  for (std::size_t p = 0; p != platform_vec.size(); ++p) {
524  try {
525  std::string name;
526  platform_vec[p].getInfo(CL_PLATFORM_NAME, &name);
527  if (setup_default_.check_platform(name)) {
528  platform_ = platform_vec[p];
529  return true;
530  }
531  } catch (cl::Error) {}
532  }
533  }
534 
535  // Using default platform: finding the first that has the device type
536  for (std::size_t p = 0; p != platform_vec.size(); ++p) {
537  try {
538  std::vector<cl::Device> dev_pool;
539  std::vector<cl::Device> dev_select;
540  platform_vec[p].getDevices(dev_type, &dev_pool);
541  device_filter(dev_pool, dev_select);
542  if (dev_select.size() != 0) {
543  platform_ = platform_vec[p];
544  return true;
545  }
546  } catch (cl::Error) {}
547  }
548 
549  return false;
550  }

552  void device_filter (const std::vector<cl::Device> &dev_pool,
553  std::vector<cl::Device> &dev_select)
554  {
555  std::vector<bool> dev_select_idx(dev_pool.size(), true);
556 
557  // Not using the default device vendor
558  if (!setup_default_.default_device_vendor()) {
559  for (std::size_t d = 0; d != dev_pool.size(); ++d) {

561  std::string str;

563  if (!setup_default_.check_device_vendor(str))
564  dev_select_idx[d] = false;
565  } catch (cl::Error) {
566  dev_select_idx[d] = false;
567  }
568  }
569  }
570 
571  // Not using the default device
572  if (!setup_default_.default_device()) {
573  for (std::size_t d = 0; d != dev_pool.size(); ++d) {
574  try {
575  std::string str;
576  dev_pool[d].getInfo(CL_DEVICE_NAME, &str);
577  if (!setup_default_.check_device(str))
578  dev_select_idx[d] = false;
579  } catch (cl::Error) {
580  dev_select_idx[d] = false;
581  }
582  }
583  }
584 
585  for (std::size_t d = 0; d != dev_pool.size(); ++d) {
586  if (dev_select_idx[d]) {
587  try {
588  dev_select.push_back(dev_pool[d]);
589  } catch (cl::Error) {}
590  }
591  }
