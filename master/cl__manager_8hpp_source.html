454  void setup_cl_manager (cl_device_type dev_type)
455  {
456  setup_ = false;
457 
458  bool setup_platform = platform_filter(dev_type);
459  VSMC_RUNTIME_WARNING_CL_MANAGER_SETUP_PLATFORM;
460  if (!setup_platform) return;
461 
462  bool setup_context = false;
463  bool setup_device = false;
464  try {
465  std::vector< ::cl::Device> dev_pool;
466  std::vector< ::cl::Device> dev_select;
467  platform_.getDevices(dev_type, &dev_pool);
468  device_filter(dev_pool, dev_select);
469  if (dev_select.size() != 0) {
470  cl_context_properties context_properties[] = {
471  static_cast<cl_context_properties>(CL_CONTEXT_PLATFORM),
472  reinterpret_cast<cl_context_properties>(platform_()), 0
473  };
474  context_ = ::cl::Context(dev_select, context_properties);
475  setup_context = true;
476  device_vec_ = context_.getInfo<CL_CONTEXT_DEVICES>();
477  device_ = device_vec_[0];
478  setup_device = true;
479  }
480  } catch (::cl::Error) {}
481  VSMC_RUNTIME_WARNING_CL_MANAGER_SETUP_CONTEXT;
482  VSMC_RUNTIME_WARNING_CL_MANAGER_SETUP_DEVICE;
483  if (!setup_context) return;
484  if (!setup_device) return;
485 
486  bool setup_command_queue = false;
487  try {
488  command_queue_ = ::cl::CommandQueue(context_, device_, 0);
489  setup_command_queue = true;
490  } catch (::cl::Error) {}
491  VSMC_RUNTIME_WARNING_CL_MANAGER_SETUP_COMMAND_QUEUE;
492  if (!setup_command_queue) return;
493 
494  setup_ = true;
495  }

497  bool platform_filter (cl_device_type dev_type)
498  {
499  std::vector< ::cl::Platform> platform_vec;
500  try {
501  ::cl::Platform::get(&platform_vec);
502  } catch (::cl::Error) {
503  platform_vec.clear();
504  }
505  if (platform_vec.size() == 0)
506  return false;
507 
508  // If not using default platform
509  if (!setup_default_.default_platform()) {
510  for (std::size_t p = 0; p != platform_vec.size(); ++p) {
511  try {
512  std::string name;
513  platform_vec[p].getInfo(CL_PLATFORM_NAME, &name);
514  if (setup_default_.check_platform(name)) {
515  platform_ = platform_vec[p];
516  return true;
517  }
518  } catch (::cl::Error) {}
519  }
520  }
521 
522  // Using default platform: finding the first that has the device type
523  for (std::size_t p = 0; p != platform_vec.size(); ++p) {
524  try {
525  std::vector< ::cl::Device> dev_pool;
526  std::vector< ::cl::Device> dev_select;
527  platform_vec[p].getDevices(dev_type, &dev_pool);
528  device_filter(dev_pool, dev_select);
529  if (dev_select.size() != 0) {
530  platform_ = platform_vec[p];
531  return true;
532  }
533  } catch (::cl::Error) {}
534  }
535 
536  return false;
537  }

540  std::vector< ::cl::Device> &dev_select)
541  {
542  std::vector<bool> dev_select_idx(dev_pool.size(), true);

551  dev_select_idx[d] = false;

553  dev_select_idx[d] = false;
554  }
555  }
556  }
557 
558  // Not using the default device

560  for (std::size_t d = 0; d != dev_pool.size(); ++d) {
561  try {
562  std::string str;

564  if (!setup_default_.check_device(str))
565  dev_select_idx[d] = false;

567  dev_select_idx[d] = false;
568  }

570  }
