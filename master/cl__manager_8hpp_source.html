389  void setup_cl_manager (cl_device_type dev_type)
390  {
391  setup_ = false;
392 
393  bool setup_platform = platform_filter(dev_type);
394  VSMC_RUNTIME_WARNING_CL_MANAGER_SETUP_PLATFORM;
395  if (!setup_platform) return;
396 
397  bool setup_context = false;
398  bool setup_device = false;
399  try {
400  std::vector< ::cl::Device> dev_pool;
401  std::vector< ::cl::Device> dev_select;
402  platform_.getDevices(dev_type, &dev_pool);
403  device_filter(dev_pool, dev_select);
404  if (dev_select.size() != 0) {
405  cl_context_properties context_properties[] = {
406  static_cast<cl_context_properties>(CL_CONTEXT_PLATFORM),
407  reinterpret_cast<cl_context_properties>(platform_()), 0
408  };
409  context_ = ::cl::Context(dev_select, context_properties);
410  setup_context = true;
411  device_vec_ = context_.getInfo<CL_CONTEXT_DEVICES>();
412  device_ = device_vec_[0];
413  setup_device = true;
414  }
415  } catch (::cl::Error) {}
416  VSMC_RUNTIME_WARNING_CL_MANAGER_SETUP_CONTEXT;
417  VSMC_RUNTIME_WARNING_CL_MANAGER_SETUP_DEVICE;
418  if (!setup_context) return;
419  if (!setup_device) return;
420 
421  bool setup_command_queue = false;
422  try {
423  command_queue_ = ::cl::CommandQueue(context_, device_, 0);
424  setup_command_queue = true;
425  } catch (::cl::Error) {}
426  VSMC_RUNTIME_WARNING_CL_MANAGER_SETUP_COMMAND_QUEUE;
427  if (!setup_command_queue) return;
428 
429  setup_ = true;
430  }

432  bool platform_filter (cl_device_type dev_type)
433  {
434  std::vector< ::cl::Platform> platform_vec;
435  try {
436  ::cl::Platform::get(&platform_vec);
437  } catch (::cl::Error) {
438  platform_vec.clear();
439  }
440  if (platform_vec.size() == 0)
441  return false;
442 
443  // If not using default platform
444  if (!setup_default_.default_platform()) {
445  for (std::size_t p = 0; p != platform_vec.size(); ++p) {
446  try {
447  std::string name;
448  platform_vec[p].getInfo(CL_PLATFORM_NAME, &name);
449  if (setup_default_.check_platform(name)) {
450  platform_ = platform_vec[p];
451  return true;
452  }
453  } catch (::cl::Error) {}
454  }
455  }
456 
457  // Using default platform: finding the first that has the device type
458  for (std::size_t p = 0; p != platform_vec.size(); ++p) {
459  try {
460  std::vector< ::cl::Device> dev_pool;
461  std::vector< ::cl::Device> dev_select;
462  platform_vec[p].getDevices(dev_type, &dev_pool);
463  device_filter(dev_pool, dev_select);
464  if (dev_select.size() != 0) {
465  platform_ = platform_vec[p];
466  return true;
467  }
468  } catch (::cl::Error) {}
469  }
470 
471  return false;
472  }

474  void device_filter (const std::vector< ::cl::Device> &dev_pool,
475  std::vector< ::cl::Device> &dev_select)
476  {
477  std::vector<bool> dev_select_idx(dev_pool.size(), true);
478 
479  // Not using the default device vendor
480  if (!setup_default_.default_device_vendor()) {
481  for (std::size_t d = 0; d != dev_pool.size(); ++d) {
482  try {
483  std::string str;
484  dev_pool[d].getInfo(CL_DEVICE_VENDOR, &str);
485  if (!setup_default_.check_device_vendor(str))
486  dev_select_idx[d] = false;
487  } catch (::cl::Error) {
488  dev_select_idx[d] = false;
489  }
490  }
491  }
492 
493  // Not using the default device
494  if (!setup_default_.default_device()) {
495  for (std::size_t d = 0; d != dev_pool.size(); ++d) {
496  try {
497  std::string str;
498  dev_pool[d].getInfo(CL_DEVICE_NAME, &str);
499  if (!setup_default_.check_device(str))
500  dev_select_idx[d] = false;
501  } catch (::cl::Error) {
502  dev_select_idx[d] = false;
503  }
504  }
505  }
506 
507  for (std::size_t d = 0; d != dev_pool.size(); ++d) {
508  if (dev_select_idx[d]) {
509  try {
510  dev_select.push_back(dev_pool[d]);
511  } catch (::cl::Error) {}
512  }
513  }
514  }
