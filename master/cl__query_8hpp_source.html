358  template <typename CharT, typename Traits>
359  static void print_val (std::basic_ostream<CharT, Traits> &os,
360  const std::string &val)
361  {
362  if (val.size() <= 60) {
363  os << val;
364  return;
365  }
366 
367  std::string::size_type pos = val.find(' ');
368  if (pos == std::string::npos) {
369  os << val;
370  return;
371  }
372 
373  std::string::size_type init = 0;
374  std::vector<std::string> val_tmp;
375  std::vector<std::string> val_vec;
376  while (pos != std::string::npos) {
377  val_tmp.push_back(val.substr(init, pos - init + 1));
378  init = pos + 1;
379  pos = val.find(' ', init);
380  }
381  for (std::size_t i = 0; i != val_tmp.size(); ++i) {
382  if (val_tmp[i] != std::string(" "))
383  val_vec.push_back(val_tmp[i]);
384  }
385 
386  if (val_vec.size() == 0)
387  return;
388 
389  if (val_vec.size() == 1) {
390  os << val_vec[0];
391  return;
392  }
393 
394  os << val_vec[0] << '\n';
395  for (std::size_t i = 1; i != val_vec.size() - 1; ++i) {
396  for (std::size_t s = 0; s != 41; ++s)
397  os << ' ';
398  os << val_vec[i] << '\n';
399  }
400  for (std::size_t i = 0; i != 41; ++i)
401  os << ' ';
402  os << val_vec.back();
403  }
