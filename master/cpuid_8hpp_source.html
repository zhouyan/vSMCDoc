366  {
367  int CPUInfo[4] = {0};

369  std::memcpy(&InfoType[0], &eax, sizeof(int));
370  std::memcpy(&InfoType[1], &ecx, sizeof(int));
371  __cpuidex(CPUInfo, InfoType[0], InfoType[1]);
372  Array<unsigned, 4> reg;

386  "cpuid\n"

390  Array<unsigned, 4> reg;

392  reg.at<1>() = ebx;

409  static std::string vendor_id ()
410  {
411  Array<unsigned, 4> reg(info(0x00U));
412  const unsigned *uptr = reg.data();
413  char str[sizeof(unsigned) * 3 + 1] = {'\0'};
414  std::memcpy(str + sizeof(unsigned) * 0, uptr + 1, sizeof(unsigned));
415  std::memcpy(str + sizeof(unsigned) * 1, uptr + 3, sizeof(unsigned));
416  std::memcpy(str + sizeof(unsigned) * 2, uptr + 2, sizeof(unsigned));
417 
418  return std::string(str);
419  }

625  template <typename CharT, typename Traits>
626  static void print_strvec (std::basic_ostream<CharT, Traits> &os,
627  const std::vector<std::string> &strvec, std::size_t fix = 20)
628  {
629  const std::size_t N = strvec.size();
630  std::size_t rows = N / 4;
631  if (N % 4 != 0)
632  ++rows;
633  for (std::size_t r = 0; r != rows; ++r) {
634  std::size_t index;
635  if ((index = r + rows * 0) < N) print_str(os, strvec[index], fix);
636  if ((index = r + rows * 1) < N) print_str(os, strvec[index], fix);
637  if ((index = r + rows * 2) < N) print_str(os, strvec[index], fix);
638  if ((index = r + rows * 3) < N) print_str(os, strvec[index], fix);
639  os << '\n';
640  }
641  }

645  const std::string &str, std::size_t fix = 20)

648  if (str.size() < fix) {
