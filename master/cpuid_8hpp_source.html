714  template <typename CharT, typename Traits>
715  static void print_feature (std::basic_ostream<CharT, Traits> &os,
716  std::vector<std::string> &feats)
717  {
718  std::sort(feats.begin(), feats.end());
719  const std::size_t N = feats.size();
720  const std::size_t fix = 15;
721  const std::size_t cols = 6;
722  std::size_t rows = N / cols + (N % cols == 0 ? 0 : 1);
723  std::size_t index = 0;
724  for (std::size_t r = 0; r != rows; ++r) {
725  for (std::size_t c = 0; c != cols; ++c)
726  if ((index = r + rows * c) < N)
727  print_feat(os, feats[index], fix);
728  os << '\n';
729  }
730  }
