182 inline InputIter hdf5store_matrix (std::size_t nrow, std::size_t ncol,
183  const std::string &file_name, const std::string &data_name,
184  InputIter first, bool append = false)
185 {
186  if (nrow == 0 || ncol == 0)
187  return first;
188 
189  std::string dataset_name("/" + data_name);
190  ::hsize_t dim[2];
191  internal::hdf5io_matrix_dim<Order>(nrow, ncol, dim);
192  internal::HDF5DataPtr<T> data_ptr;
193  InputIter last = data_ptr.set(nrow * ncol, first);
194  const T *data = data_ptr.get();
195 
196  ::hid_t datafile;
197  if (append) {
198  datafile = ::H5Fopen(file_name.c_str(), H5F_ACC_RDWR, H5P_DEFAULT);
199  } else {
200  datafile = ::H5Fcreate(file_name.c_str(),
201  H5F_ACC_TRUNC, H5P_DEFAULT, H5P_DEFAULT);
202  }
203  ::hid_t dataspace = ::H5Screate_simple(2, dim, VSMC_NULLPTR);
204  ::hid_t datatype = hdf5io_datatype<T>();
205  ::hid_t dataset = ::H5Dcreate(datafile, dataset_name.c_str(),
206  datatype, dataspace, H5P_DEFAULT, H5P_DEFAULT, H5P_DEFAULT);
207  ::H5Dwrite(dataset, datatype, H5S_ALL, H5S_ALL, H5P_DEFAULT, data);
208 
209  ::H5Dclose(dataset);
210  ::H5Tclose(datatype);
211  ::H5Sclose(dataspace);
212  ::H5Fclose(datafile);
213 
214  return last;

458  data_ptr[j] = &data[j * nrow];

461  data_ptr.begin(), header.begin(), append);
