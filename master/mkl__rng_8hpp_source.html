294  void operator() (MKLStream<BRNG> &stream, size_type nskip)
295  {
296  if (nskip == 0)

299  if (nskip < buffer_size_) {
300  if (buffer_.size() < nskip)

308  size_type n = r * buffer_size_;
309  uniform_bits_(stream, n, &buffer_[0]);
310  }
311  uniform_bits_(stream, remain, &buffer_[0]);
312  }
313  }
