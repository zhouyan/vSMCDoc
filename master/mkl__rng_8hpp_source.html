309  size_type remain = nskip - repeat * buffer_size_;
310  for (size_type r = 1; r != repeat + 1; ++r) {
311  size_type n = r * buffer_size_;
312  uniform_bits_(stream, n, &buffer_[0]);
313  }
314  uniform_bits_(stream, remain, &buffer_[0]);
315  }
316  }
