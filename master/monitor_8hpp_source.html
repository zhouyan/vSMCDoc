260  void eval (std::size_t iter, const Particle<T> &particle)
261  {
262  if (!recording_)
263  return;
264 
265  VSMC_RUNTIME_ASSERT_CORE_MONITOR_FUNCTOR(eval_, eval, EVALUATION);
266 
267  const std::size_t N = static_cast<std::size_t>(particle.size());
268  double *const result = malloc_result(dim_);
269  double *const buffer = malloc_eval_integrand(N * dim_);
270  double *const weight = malloc_weight(N);
271  particle.read_weight(weight);
272 
273  eval_(iter, dim_, particle, buffer);
274  if (dim_ == 1) {
275  double res = 0;
276  for (std::size_t i = 0; i != N; ++i)
277  res += buffer[i] * weight[i];
278  result[0] = res;
279  } else {
