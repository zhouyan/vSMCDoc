425  double operator() (double alpha)
426  {
427  using std::exp;
428 
429  if (path_.iter_size() == 0)
430  return 0;
431 
432  std::size_t iter = 0;
433  while (iter != path_.iter_size() && path_.grid(iter) < alpha)
434  ++iter;
435 
436  double tol = alpha * rel_err_;
437  tol = tol < abs_err_ ? tol : abs_err_;
438  tol = tol > 0 ? tol : abs_err_;
439 
440  if (iter != path_.iter_size() && path_.grid(iter) - alpha < tol)
441  return path_.integrand(iter);
442 
443  if (iter == 0)
444  return 0;
445 
446  --iter;
447  if (alpha - path_.grid(iter) < tol)
448  return path_.integrand(iter);

451  std::size_t size = weight_.size();
