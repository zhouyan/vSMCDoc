176 template <std::size_t K, std::size_t I>
177 inline void philox_hilo (uint64_t b, uint64_t &hi, uint64_t &lo)
178 {
179  const uint64_t a =
180  traits::PhiloxRoundConstantTrait<uint64_t, K, I>::value;
181  const unsigned whalf = 32;
182  const uint64_t lomask = (static_cast<uint64_t>(1) << whalf) - 1;
183 
184  lo = static_cast<uint64_t>(a * b);
185 
186  const uint64_t ahi = a >> whalf;
187  const uint64_t alo = a & lomask;
188  const uint64_t bhi = b >> whalf;
189  const uint64_t blo = b & lomask;
190 
191  const uint64_t ahbl = ahi * blo;
192  const uint64_t albh = alo * bhi;
193 
194  const uint64_t ahbl_albh = ((ahbl & lomask) + (albh & lomask));
195 
196  hi = ahi * bhi + (ahbl >> whalf) + (albh >> whalf);
197  hi += ahbl_albh >> whalf;
198  hi += ((lo >> whalf) < (ahbl_albh & lomask));
199 }
