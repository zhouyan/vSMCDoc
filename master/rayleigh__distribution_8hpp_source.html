93 template <typename RealType, typename RNGType>

96 {

100 
101  const std::size_t k = 1024;
102  const std::size_t m = n / k;
103  const std::size_t l = n % k;
104  for (std::size_t i = 0; i != m; ++i, r += k)
105  internal::rayleigh_distribution_impl(rng, k, r, sigma);
106  internal::rayleigh_distribution_impl(rng, l, r, sigma);
107 }
