134  template <typename Range>
135  std::vector<Range> partition (const Range &range) const
136  {
137  typedef typename Range::const_iterator size_type;
138  size_type N = range.end() - range.begin();
139  size_type tn = static_cast<size_type>(thread_num());
140  size_type block_size = 0;
141 
142  if (N < tn)
143  block_size = 1;
144  else if (N % tn)
145  block_size = N / tn + 1;
146  else
147  block_size = N / tn;
148 
149  std::vector<Range> range_vec;
150  range_vec.reserve(thread_num());
151  size_type B = range.begin();
152  while (N > 0) {
153  size_type next = N < block_size ? N : block_size;
154  range_vec.push_back(Range(B, B + next));
155  B += next;
156  N -= next;
157  }
158 
159  return range_vec;
160  }
