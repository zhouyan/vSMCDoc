450  virtual double compute_ess (const double *first, bool use_log) const
451  {
452  std::vector<double> buffer(size_);
453  double *const bptr = &buffer[0];
454 
455  if (use_log) {
456  math::vAdd(size_, &log_weight_[0], first, bptr);
457  double dmax = bptr[0];
458  for (size_type i = 0; i != size_; ++i)
459  if (dmax < bptr[i])
460  dmax = bptr[i];
461  dmax = -dmax;
462  for (size_type i = 0; i != size_; ++i)
463  bptr[i] += dmax;
464  math::vExp(size_, bptr, bptr);
465  } else {
466  math::vMul(size_, &weight_[0], first, bptr);
467  }
468 
469  double coeff = 1 / math::asum(size_, bptr);
470  math::scal(size_, coeff, bptr);
471 
472  return 1 / math::dot(size_, bptr, bptr);
473  }

476  virtual double compute_cess (const double *first, bool use_log) const
477  {
478  const double *bptr = first;
479  const double *const wptr = &weight_[0];
480  std::vector<double> buffer;
481  if (use_log) {
482  buffer.resize(size_);
483  math::vExp(size_, first, &buffer[0]);
484  bptr = &buffer[0];
485  }
486 
487  double above = 0;
488  double below = 0;
489  for (size_type i = 0; i != size_; ++i) {
490  double wb = wptr[i] * bptr[i];
491  above += wb;
492  below += wb * bptr[i];
493  }
494 
495  return above * above / below;
