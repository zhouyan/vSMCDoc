52 template <typename Range, typename T, typename WorkType>
53 inline T parallel_accumulate (const Range &range, WorkType &&work, T init)
54 {
55  std::vector<Range> range_vec(ThreadNum::instance().partition(range));
56  std::vector<T> result(range_vec.size());
57  // start parallelization
58  {
59  std::vector<ThreadGuard<std::thread>> tg;
60  tg.reserve(range_vec.size());
61  for (std::size_t i = 0; i != range_vec.size(); ++i) {
62  tg.push_back(ThreadGuard<std::thread>(std::thread(
63  std::forward<WorkType>(work),
64  range_vec[i], std::ref(result[i]))));
65  }
66  }
67  // stop parallelization
68  T acc(init);
69  for (std::size_t i = 0; i != result.size(); ++i)
70  acc += result[i];
