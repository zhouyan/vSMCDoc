385  virtual double compute_ess (const double *first, bool use_log) const
386  {
387  std::vector<double, AlignedAllocator<double> > buffer(size_);
388  double *const bptr = &buffer[0];
389 
390  if (use_log) {
391  math::vAdd(size_, &log_weight_[0], first, bptr);
392  double dmax = bptr[0];
393  for (size_type i = 0; i != size_; ++i)
394  if (dmax < bptr[i])
395  dmax = bptr[i];
396  dmax = -dmax;
397  for (size_type i = 0; i != size_; ++i)
398  bptr[i] += dmax;
399  math::vExp(size_, bptr, bptr);
400  } else {
401  math::vMul(size_, &weight_[0], first, bptr);
402  }
403 
404  double coeff = 1 / math::asum(size_, bptr);
405  math::scal(size_, coeff, bptr);
406 
407  return 1 / math::dot(size_, bptr, bptr);
408  }

411  virtual double compute_cess (const double *first, bool use_log) const
412  {
413  const double *bptr = first;
414  const double *const wptr = &weight_[0];
415  std::vector<double, AlignedAllocator<double> > buffer;
416  if (use_log) {
417  buffer.resize(size_);
418  math::vExp(size_, first, &buffer[0]);
419  bptr = &buffer[0];
420  }
421 
422  double above = 0;
423  double below = 0;
424  for (size_type i = 0; i != size_; ++i) {
425  double wb = wptr[i] * bptr[i];
426  above += wb;
427  below += wb * bptr[i];
428  }
429 
430  return above * above / below;
431  }
