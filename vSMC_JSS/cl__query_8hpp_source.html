329  template <typename CharT, typename Traits>
330  static void print_val (std::basic_ostream<CharT, Traits> &os,
331  const std::string &val)
332  {
333  if (val.size() <= 60) {
334  os << val;
335  return;
336  }
337 
338  std::string::size_type pos = val.find(' ');
339  if (pos == std::string::npos) {
340  os << val;
341  return;
342  }
343 
344  std::string::size_type init = 0;
345  std::vector<std::string> val_tmp;
346  std::vector<std::string> val_vec;
347  while (pos != std::string::npos) {
348  val_tmp.push_back(val.substr(init, pos - init + 1));
349  init = pos + 1;
350  pos = val.find(' ', init);
351  }
352  for (std::size_t i = 0; i != val_tmp.size(); ++i) {
353  if (val_tmp[i] != std::string(" "))
354  val_vec.push_back(val_tmp[i]);
355  }
356 
357  if (val_vec.size() == 0)
358  return;
359 
360  if (val_vec.size() == 1) {
361  os << val_vec[0];
362  return;
363  }
364 
365  os << val_vec[0] << '\n';
366  for (std::size_t i = 1; i != val_vec.size() - 1; ++i) {
367  for (std::size_t s = 0; s != 41; ++s)
368  os << ' ';
369  os << val_vec[i] << '\n';
370  }
371  for (std::size_t i = 0; i != 41; ++i)
372  os << ' ';
373  os << val_vec.back();
374  }
